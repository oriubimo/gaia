210 static void BM_ChannelConnection(benchmark::State& state) {
211  IoContextPool pool(1);
212  pool.Run();
213  AcceptServer server(&pool);
214 
215  TestInterface ti;
216  uint16_t port = server.AddListener(0, &ti);
217 
218  server.Run();
219 
220  FiberSyncSocket socket("localhost", std::to_string(port), &pool.GetNextContext());
221  CHECK(!socket.ClientWaitToConnect(500));
222  Frame frame;
223  std::string send_msg(100, 'a');
224  uint8_t buf[Frame::kMaxByteSize];
225  BufferType header, letter;
226  frame.letter_size = send_msg.size();
227  letter.resize(frame.letter_size);
228 
229  uint64_t fs = frame.Write(buf);
230  BufferType tmp;
231  tmp.resize(10000);
232 
233  auto buf_seq = make_buffer_seq(asio::buffer(buf, fs), send_msg);
234  size_t total_sz = 0, buf_seq_sz = asio::buffer_size(buf_seq);
235 
236  socket.context().AwaitSafe([&] {
237  system::error_code ec;
238  while (state.KeepRunning()) {
239  asio::write(socket, buf_seq, ec);
240  CHECK(!ec);
241  total_sz += buf_seq_sz;
242 
243  DCHECK(!ec);
244  DCHECK_EQ(0, frame.header_size);
245  DCHECK_EQ(letter.size(), frame.letter_size);
246  socket.read_some(asio::buffer(tmp), ec);
247  }
248  });
249 
250  state.SetBytesProcessed(total_sz);
251  server.Stop();
252 }
