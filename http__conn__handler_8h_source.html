53  template <typename Body>
54  void Invoke(Response<Body>&& msg) {
55  // Determine if we should close the connection after
56  // close_ = msg.need_eof();
57 
58  // We need the serializer here because the serializer requires
59  // a non-const file_body, and the message oriented version of
60  // http::write only works with const messages.
61  msg.prepare_payload();
62  ::boost::beast::http::response_serializer<Body> sr{msg};
63 
64  ::boost::beast::http::write(stream_, sr, ec);
65  }
