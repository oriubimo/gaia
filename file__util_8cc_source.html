61 static void TraverseRecursivelyInternal(StringPiece path, std::function<void(StringPiece)> cb,
62  uint32 offset) {
63  struct stat stats;
64  if (stat(path.data(), &stats) != 0) {
65  LOG(ERROR) << "TraverseRecursively failed for " << path << " with error " << error_str(errno);
66  return;
67  }
68 
69  if (S_ISDIR(stats.st_mode)) {
70  string current_name;
71  DIR* dir = opendir(path.data());
72  if (dir == NULL) {
73  LOG(ERROR) << "TraverseRecursively: error opening dir " << path
74  << ", error: " << error_str(errno);
75  return;
76  }
77 
78  while (true) {
79  struct dirent* der = readdir(dir);
80  if (der == nullptr)
81  break;
82  StringPiece entry_name(der->d_name);
83  if (entry_name == StringPiece(".") || entry_name == StringPiece("..")) {
84  continue;
85  }
86  current_name = JoinPath(path, entry_name);
87  if (der->d_type == DT_DIR) {
88  TraverseRecursivelyInternal(current_name, cb, offset);
89  } else {
90  StringPiece cn(current_name);
91  cn.remove_prefix(offset);
92  cb(cn);
93  }
94  }
95  closedir(dir);
96  } else if (S_ISREG(stats.st_mode)) {
97  path.remove_prefix(offset);
98  cb(path);
99  } else {
100  LOG(WARNING) << "unknown type " << stats.st_mode;
101  }
102 }

186 bool RecursivelyCreateDir(StringPiece path, int mode) {
187  struct stat st;
188  int res = stat(path.data(), &st);
189  if (res == 0 && (st.st_mode & S_IFMT) == S_IFDIR){
190  return true;
191  }
192 
193  if (CreateDir(path, mode))
194  return true;
195 
196  string str(path);
197 
198  // Try creating the parent.
199  string::size_type slashpos = str.rfind('/');
200  if (slashpos == string::npos) {
201  // No parent given.
202  return false;
203  }
204  str.resize(slashpos);
205 
206  return RecursivelyCreateDir(str, mode) && CreateDir(path, mode);
207 }

209 void DeleteRecursively(StringPiece name) {
210  // We don't care too much about error checking here since this is only used
211  // in tests to delete temporary directories that are under /tmp anyway.
212 
213  // Use opendir()! Yay!
214  // lstat = Don't follow symbolic links.
215  struct stat stats;
216  if (lstat(name.data(), &stats) != 0)
217  return;
218 
219  if (S_ISDIR(stats.st_mode)) {
220  DIR* dir = opendir(name.data());
221  if (dir != NULL) {
222  while (true) {
223  struct dirent* entry = readdir(dir);
224  if (entry == NULL)
225  break;
226  string entry_name = entry->d_name;
227  if (entry_name != "." && entry_name != "..") {
228  DeleteRecursively(StrCat(name, "/", entry_name));
229  }
230  }
231  }
232 
233  closedir(dir);
234  rmdir(name.data());
235 
236  } else if (S_ISREG(stats.st_mode)) {
237  remove(name.data());
238  }
239 }

284 bool TempFile::TempFilename(const char* directory_prefix, string* filename) {
285  CHECK(filename != NULL);
286  filename->clear();
287 
288  if (directory_prefix != NULL) {
289  TempFilenameInDir(directory_prefix, filename);
290  return true;
291  }
292 
293  // Re-fetching available dirs ensures thread safety.
294  vector<string> dirs;
295  // TODO(tkaftal): Namespace might vary depending on glog installation options,
296  // I should probably use a configure/make flag here.
297  google::GetExistingTempDirectories(&dirs);
298 
299  // Try each directory, as they might be full, have inappropriate
300  // permissions or have different problems at times.
301  for (size_t i = 0; i < dirs.size(); ++i) {
302  TempFilenameInDir(dirs[i].c_str(), filename);
303  if (file::Exists(*filename)) {
304  LOG(WARNING) << "unique tempfile already exists in " << *filename;
305  filename->clear();
306  } else {
307  return true;
308  }
309  }
310 
311  LOG(ERROR) << "Couldn't find a suitable TempFile anywhere. Tried " << dirs.size()
312  << " directories";
313  return false;
314 }

352 // Similar to ExpandFiles but also returns statistics about files sizes and timestamps.
353 Status StatFilesSafe(StringPiece path, std::vector<StatShort>* res) {
354  CHECK_NOTNULL(res);
355  glob_t glob_result;
356  int rv = glob(path.data(), GLOB_TILDE_CHECK, errfunc, &glob_result);
357  if (rv && rv != GLOB_NOMATCH) {
358  return Status(StatusCode::IO_ERROR, strerror(rv));
359  }
360 
361  struct stat statbuf;
362  for (size_t i = 0; i < glob_result.gl_pathc; i++) {
363  if (stat(glob_result.gl_pathv[i], &statbuf) == 0) {
364  StatShort sshort{glob_result.gl_pathv[i], statbuf.st_mtime, statbuf.st_size,
365  statbuf.st_mode};
366  res->emplace_back(std::move(sshort));
367  } else {
368  LOG(WARNING) << "Bad stat for " << glob_result.gl_pathv[i];
369  }
370  }
371  globfree(&glob_result);
372  return Status::OK;
373 }

382 void CompressToGzip(StringPiece file, uint8_t compress_level) {
383  util::StatusObject<file::ReadonlyFile*> src = file::ReadonlyFile::Open(file);
384  CHECK(src.ok()) << file;
385 
386  string wfile = StrCat(file, ".gz");
387 
388  gzFile dest = gzopen(wfile.c_str(), "wb");
389  gzbuffer(dest, 1 << 16);
390  gzsetparams(dest, compress_level, Z_DEFAULT_STRATEGY);
391 
392  constexpr unsigned kBufSize = 1 << 15;
393  std::unique_ptr<uint8[]> buf(new uint8[kBufSize]);
394  size_t offs = 0, read = kBufSize;
395  std::unique_ptr<ReadonlyFile> src_file(src.obj);
396  while (read == kBufSize) {
397  auto res = src_file->Read(offs, strings::MutableByteRange(buf.get(), kBufSize));
398  CHECK(res.ok()) << res.status;
399 
400  read = res.obj;
401  if (res.obj) {
402  CHECK_GT(gzwrite(dest, buf.get(), read), 0);
403  }
404  offs += read;
405  }
406  src_file->Close();
407  gzclose_w(dest);
408  CHECK(file::Delete(file));
409 }

411 void CopyFileOrDie(StringPiece src, StringPiece dest_path) {
412  int source = open(src.data(), O_RDONLY, 0);
413 
414  CHECK_GT(source, 0);
415 
416  // struct required, rationale: function stat() exists also
417  struct stat stat_source, stat_dest;
418  int err = fstat(source, &stat_source);
419  CHECK_EQ(0, err);
420 
421  string dest_file;
422  if (stat(dest_path.data(), &stat_dest) == 0) {
423  if (S_ISDIR(stat_dest.st_mode)) {
424  dest_file = JoinPath(dest_path, GetNameFromPath(src));
425  dest_path = dest_file.c_str();
426  }
427  }
428  int dest = open(dest_path.data(), O_WRONLY | O_CREAT | O_TRUNC, 0644);
429  CHECK_GT(dest, 0);
430 
431  CHECK_GE(sendfile(dest, source, 0, stat_source.st_size), 0);
432 
433  close(source);
434  close(dest);
435 }
