79 unsigned SingleProducerTaskPoolBase::FindMostFreeThread() const {
80  // Give each thread a score according to his queue size and if its runnning task.
81  // Rerun thread index with lowest score.
82  uint32 min_score = kuint32max;
83  unsigned index = 0;
84  for (unsigned i = 0; i < thread_count_; ++i) {
85  uint32 score = thread_interfaces_[i]->QueueSize();
86  if (thread_info_[i].d.has_tasks) {
87  ++score;
88  }
89  if (score == 0) {
90  return i; // If found thread with score 0 return this thread index.
91  }
92  if (score < min_score) {
93  index = i;
94  min_score = score;
95  }
96  }
97  return index;
98 }

101 void SingleProducerTaskPoolBase::WaitForTasksToComplete() {

134 void* SingleProducerTaskPoolBase::ThreadRoutine(void* arg) {
135  RoutineConfig* config = (RoutineConfig*)arg;
136  SingleProducerTaskPoolBase* me = config->me;
137  ThreadInfo::Data& ti = me->thread_info_[config->thread_index].d;
138 
139  ThreadLocalInterface* thread_interface = me->thread_interfaces_[config->thread_index].get();
140 
141  delete config;
142  config = nullptr;
143  auto await_check = [me, thread_interface]() {
144  return me->start_cancel_ || !thread_interface->IsQueueEmpty();
145  };
146 
147  unsigned num_yields = 0;
148  while (!me->start_cancel_) {
149  ti.has_tasks.store(true, std::memory_order_release);
150  while (thread_interface->RunTask()) {
151  num_yields = 0;
152  }
153  ti.has_tasks.store(false, std::memory_order_release);
154 
155  if (++num_yields > 100) {
156  // We iterate 1000 times before even bother to sleep.
157 
158  if (thread_interface->IsQueueEmpty()) {
159  VLOG(2) << "ti.empty_q_cv.notify";
160 
161  ti.ev_task_finished.notify();
162  ti.ev_non_empty.await(await_check);
163  }
164  }
165  }
166  char buf[30] = {0};
167  pthread_getname_np(pthread_self(), buf, sizeof buf);
168  VLOG(1) << "Finishing running SingleProducerTaskPoolBase thread " << buf;
169 
170  return NULL;
171 }
