62  friend void intrusive_ptr_release(ConnectionHandler* ctx) noexcept {
63  if (1 == ctx->use_count_.fetch_sub(1, std::memory_order_release)) {
64  // We want to synchronize on all changes to ctx performed in other threads.
65  // ctx is not atomic but we know that whatever was being written - has been written
66  // in other threads and no references to ctx exist anymore.
67  // Therefore acquiring fence is enough to synchronize.
68  // "acquire" requires a release opearation to mark the end of the memory changes we wish
69  // to acquire, and "fetch_sub(std::memory_order_release)" provides this marker.
70  // To summarize: fetch_sub(release) and fence(acquire) needed to order and synchronize
71  // on changes on ctx in most performant way.
72  // See: https://stackoverflow.com/q/27751025/
73  std::atomic_thread_fence(std::memory_order_acquire);
74  delete ctx;
75  }
76  }
