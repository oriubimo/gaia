46 void IoContextPool::Run() {
47  CHECK_EQ(STOPPED, state_);
48 
49  fibers_ext::BlockingCounter bc(thread_arr_.size());
50  char buf[32];
51 
52  for (size_t i = 0; i < thread_arr_.size(); ++i) {
53  thread_arr_[i].work.emplace(asio::make_work_guard(*context_arr_[i].context_ptr_));
54  snprintf(buf, sizeof(buf), "IoPool%lu", i);
55  thread_arr_[i].tid =
56  base::StartThread(buf, [this, i, bc]() mutable { this->WrapLoop(i, &bc); });
57  cpu_set_t cps;
58  CPU_ZERO(&cps);
59  CPU_SET(i % thread::hardware_concurrency(), &cps);
60 
61  int rc = pthread_setaffinity_np(thread_arr_[i].tid, sizeof(cpu_set_t), &cps);
62  LOG_IF(WARNING, rc) << "Error calling pthread_setaffinity_np: " << strerror(rc) << "\n";
63  }
64 
65  // We can not use Await() here yet because StartLoop might not run yet and its implementation
66  // assumes internally that the first posted handler is issued from the StartLoop.
67  // Therefore we use BlockingCounter to wait for all the IO loops to start running.
68  bc.Wait();
69 
70  LOG(INFO) << "Running " << thread_arr_.size() << " io threads";
71  state_ = RUN;
72 }

74 void IoContextPool::Stop() {
75  if (state_ == STOPPED)
76  return;
77 
78  for (size_t i = 0; i < context_arr_.size(); ++i) {
79  context_arr_[i].Stop();
80  }
81 
82  for (TInfo& tinfo : thread_arr_) {
83  tinfo.work.reset();
84  }
85  VLOG(1) << "Asio Contexts has been stopped";
86 
87  for (size_t i = 0; i < thread_arr_.size(); ++i) {
88  pthread_join(thread_arr_[i].tid, nullptr);
89  VLOG(2) << "Thread " << i << " has joined";
90  }
91  state_ = STOPPED;
92 }

94 IoContext& IoContextPool::GetNextContext() {
95  // Use a round-robin scheme to choose the next io_context to use.
96  DCHECK_LT(next_io_context_, context_arr_.size());
97  uint32_t index = next_io_context_.load();
98  IoContext& io_context = context_arr_[index++];
99 
100  // Not-perfect round-robind since this function is non-transactional but it's valid.
101  if (index == context_arr_.size())
102  next_io_context_ = 0;
103  else
104  next_io_context_ = index;
105  return io_context;
106 }
