76 constexpr uint32_t kMaskDelta = 0xa282ead8ul;
77 
78 // Return a masked representation of crc.
79 //
80 // Motivation: it is problematic to compute the CRC of a string that
81 // contains embedded CRCs. Therefore we recommend that CRCs stored
82 // somewhere (e.g., in files) should be masked before being stored.
83 inline uint32_t Mask(uint32_t crc) {
84  // Rotate right by 15 bits and add a constant.
85  return ((crc >> 15) | (crc << 17)) + kMaskDelta;
86 }
87 
88 // Return the crc whose masked representation is masked_crc.
89 inline uint32_t Unmask(uint32_t masked_crc) {
90  uint32_t rot = masked_crc - kMaskDelta;
91  return ((rot >> 17) | (rot << 15));
92 }
