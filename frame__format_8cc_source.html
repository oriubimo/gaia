44 uint8_t Frame::DecodeStart(const uint8_t* src, ::boost::system::error_code& ec) {
45  if (kHeaderVal != LittleEndian::Load32(src)) {
46  ec = errc::make_error_code(errc::illegal_byte_sequence);
47  return 0;
48  }
49 
50  if (src[4] >> 4 != 0) { // version check
51  ec = errc::make_error_code(errc::illegal_byte_sequence);
52  return 0;
53  }
54  rpc_id = UNALIGNED_LOAD64(src + 4);
55  rpc_id >>= 8;
56 
57  return src[4] & 15;
58 }

65 unsigned Frame::Write(uint8* dest) const {
66  LittleEndian::Store32(dest, kHeaderVal);
67  dest += 4;
68  const uint8 msg_bytes_minus1 = SizeByteCountMinus1(letter_size);
69  const uint8 cntrl_bytes_minus1 = SizeByteCountMinus1(header_size);
70 
71  DCHECK_LT(msg_bytes_minus1, 4);
72  DCHECK_LT(cntrl_bytes_minus1, 4);
73 
74  uint64_t version = cntrl_bytes_minus1 | (msg_bytes_minus1 << 2) | (0 << 4);
75 
76  LittleEndian::Store64(dest, (rpc_id << 8) | version);
77  dest += 8;
78 
79  LittleEndian::Store32(dest, header_size);
80  dest += (cntrl_bytes_minus1 + 1);
81  LittleEndian::Store32(dest, letter_size);
82 
83  return 4 + 1 /* version */ + 7 /* rpc_id */ + cntrl_bytes_minus1 + msg_bytes_minus1 + 2;
84 }
