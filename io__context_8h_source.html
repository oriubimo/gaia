42 // Runs `f` asynchronously in io-context fiber. `f` should not block, lock on mutexes or Await.
43 // Spinlocks are ok but might cause performance degradation.
44 template <typename Func> void Async(::boost::asio::io_context& cntx, Func&& f) {
45  ::boost::asio::post(cntx, std::forward<Func>(f));
46 }
47 
48 // Similarly to Async(), runs 'f' in io_context thread, but waits for it to finish by blocking
49 // the calling fiber. `f` should not block because it runs directly from IO loop.
50 template <typename Func>
51 auto Await(::boost::asio::io_context& cntx, Func&& f) -> decltype(f()) {
52  // See Done class for more extensive documentation.
53  fibers_ext::Done done;
54  using ResultType = decltype(f());
55  detail::ResultMover<ResultType> mover;
56 
57  // Store done-ptr by value to increase the refcount while lambda is running.
58  Async(cntx, [&, f = std::forward<Func>(f), done]() mutable {
59  mover.Apply(f);
60  done.Notify();
61  });
62 
63  done.Wait();
64  return std::move(mover).get();
65 }

105  // Similar to asio_ext::Await(), but if we call Await from the context thread,

114  // Please note that this function uses Await, therefore can not be used inside Ring0
115  // (i.e. Async callbacks).
116  template <typename... Args> boost::fibers::fiber LaunchFiber(Args&&... args) {
117  ::boost::fibers::fiber fb;
118  // It's safe to use & capture since we await before returning.
119  Await([&] { fb = boost::fibers::fiber(std::forward<Args>(args)...); });
120  return fb;
121  }

123  // Runs possibly awating function 'f' safely in ContextThread and waits for it to finish,
124  // If we are in the context thread already, runs 'f' directly, otherwise
125  // runs it wrapped in a fiber. Should be used instead of 'Await' when 'f' itself
126  // awaits on something.
127  // To summarize: 'f' should not block its thread, but allowed to block its fiber.
128  template <typename Func> auto AwaitSafe(Func&& f) -> decltype(f()) {
129  if (InContextThread()) {
130  return f();
131  }
132 
133  using ResultType = decltype(f());
134  detail::ResultMover<ResultType> mover;
135  auto fb = LaunchFiber([&] { mover.Apply(std::forward<Func>(f)); });
136  fb.join();
137 
138  return std::move(mover).get();
139  }
