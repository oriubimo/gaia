15 // This class is all about reducing the contention on the producer side (notifications).
16 // We want notifications to be as light as possible, while waits are less important
17 // since they on the path of being suspended anyway. However, we also want to reduce number of
18 // spurious waits on the consumer side.
19 // This class has another wonderful property: notification thread does not need to lock mutex,
20 // which means it can be used from the io_context (ring0) fiber.

74  static bool should_switch(::boost::fibers::context* ctx, std::intptr_t expected) {
75  return ctx->twstatus.compare_exchange_strong(expected, static_cast<std::intptr_t>(-1),
76  std::memory_order_acq_rel) ||

100 inline bool EventCount::notify() noexcept {
101  uint64_t prev = val_.fetch_add(kAddEpoch, std::memory_order_acq_rel);
102 
103  if (UNLIKELY(prev & kWaiterMask)) {
104  auto* active_ctx = ::boost::fibers::context::active();
105 
106  /*
107  lk makes sure that when a waiting thread is entered the critical section in
108  EventCount::wait, it atomically checks val_ when entering the WAIT state.
109  We need it in order to make sure that cnd_.notify() is not called before the waiting
110  thread enters WAIT state and thus the notification is missed.
111  */
112  spinlock_lock_t lk{wait_queue_splk_};
113  while (!wait_queue_.empty()) {
114  auto* ctx = &wait_queue_.front();
115  wait_queue_.pop_front();
116 
117  if (should_switch(ctx, reinterpret_cast<std::intptr_t>(this))) {
118  // notify context
119  lk.unlock();
120  active_ctx->schedule(ctx);
121  break;
122  }
123  }
124 
125  return true;
126  }
127  return false;
128 }

130 inline bool EventCount::notifyAll() noexcept {
131  uint64_t prev = val_.fetch_add(kAddEpoch, std::memory_order_acq_rel);
132 
133  if (UNLIKELY(prev & kWaiterMask)) {
134  auto* active_ctx = ::boost::fibers::context::active();
135 
136  spinlock_lock_t lk{wait_queue_splk_};
137  wait_queue_t tmp;
138  tmp.swap(wait_queue_);
139  lk.unlock();
140 
141  while (!tmp.empty()) {
142  ::boost::fibers::context* ctx = &tmp.front();
143  tmp.pop_front();
144 
145  if (should_switch(ctx, reinterpret_cast<std::intptr_t>(this))) {
146  // notify context
147  active_ctx->schedule(ctx);
148  }
149  }
150  }
151 
152  return false;
153 };

155 // Atomically checks for epoch and waits on cond_var.
156 inline void EventCount::wait(uint32_t epoch) noexcept {
157  if ((val_.load(std::memory_order_acquire) >> kEpochShift) != epoch)
158  return;
159 
160  auto* active_ctx = ::boost::fibers::context::active();
161 
162  spinlock_lock_t lk{wait_queue_splk_};
163  if ((val_.load(std::memory_order_acquire) >> kEpochShift) == epoch) {
164  // atomically call lt.unlock() and block on *this
165  // store this fiber in waiting-queue
166  active_ctx->wait_link(wait_queue_);
167  active_ctx->twstatus.store(static_cast<std::intptr_t>(0), std::memory_order_release);
168 
169  // suspend this fiber
170  active_ctx->suspend(lk);
171  }
172 }

174 // Returns true if had to preempt, false if no preemption happenned.
175 template <typename Condition> bool EventCount::await(Condition condition) {
176  if (condition())
177  return false; // fast path
178 
179  // condition() is the only thing that may throw, everything else is
180  // noexcept, Key destructor makes sure to cancelWait state when exiting the function.
181  bool preempt = false;
182  while (true) {
183  Key key = prepareWait();
184  if (condition()) {
185  break;
186  }
187  preempt = true;
188  wait(key.epoch());
189  }
190  return preempt;
191 }
