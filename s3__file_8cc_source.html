131 std::pair<StringPiece, const S3BucketContext*> GetKeyAndBucket(StringPiece name) {
132  DCHECK(name.starts_with(kS3NamespacePrefix)) << name;
133  std::call_once(s3data_once, InitS3Data);
134 
135  StringPiece tmp = name;
136  tmp.remove_prefix(sizeof(kS3NamespacePrefix) - 1);
137  size_t pos = tmp.find('/');
138  CHECK_NE(StringPiece::npos, pos) << "Invalid filename " << name;
139 
140  string bucket_str = tmp.substr(0, pos).as_string();
141  BucketMap& bmap = bucket_map();
142  auto res = bmap.emplace(bucket_str, NewBucket());
143  if (res.second) {
144  res.first->second.bucketName = res.first->first.c_str();
145  }
146  tmp.remove_prefix(pos + 1);
147  return std::pair<StringPiece, const S3BucketContext*>(tmp, &res.first->second);
148 }

150 Status S3File::Read(size_t offset, size_t length, Slice* result,
151  uint8* buffer) {
152  if (offset > file_size_ || length == 0) {
153  *result = Slice();
154  return Status(StatusCode::RUNTIME_ERROR, "Invalid read range");
155  }
156  if (offset + length > file_size_) {
157  length = file_size_ - offset;
158  }
159  VLOG(1) << "S3File::Read " << offset << " " << length;
160 
161  S3GetObjectHandler handler{{nullptr, S3File::CompleteCallback},
162  S3File::DataCallback};
163  CallbackData data;
164  data.dest_buf = reinterpret_cast<char*>(buffer);
165  size_t try_length = length;
166  size_t read_length = 0;
167 
168  for (unsigned attempts = 0; attempts < 5; ++attempts) {
169  data.status = S3StatusOK;
170  data.length = 0;
171  S3_get_object(context_, key_.c_str(), nullptr, offset, try_length, nullptr, &handler, &data);
172  read_length += data.length;
173 
174  if (data.status == S3StatusOK) {
175  result->set(reinterpret_cast<char*>(buffer), read_length);
176  return Status::OK;
177  }
178  LOG(WARNING) << "Attempt " << attempts << " failed: " << data.status << ", " << data.length;
179 
180  if (data.length >= length) {
181  return Status(StatusCode::INTERNAL_ERROR, "Data length is too large");
182  }
183  offset += data.length;
184  try_length -= data.length;
185  LOG(WARNING) << "Attempt " << attempts << " failed";
186  }
187  return Status(StatusCode::IO_ERROR, S3_get_status_name(data.status));
188 }

255 Status CopyFileToS3(StringPiece file_path, StringPiece s3_dir) {
256  StringPiece file_name = file_util::GetNameFromPath(file_path);
257  if (file_name.empty()) {
258  return Status(StrCat("Invalid path: ", file_path));
259  }
260 
261  string s3_path = file_util::JoinPath(s3_dir, file_name);
262  auto res = GetKeyAndBucket(s3_path);
263  int64_t file_length = file_util::LocalFileSize(file_path);
264  if (file_length == -1) {
265  return Status(StrCat("Failed to fetch file size: ", file_path));
266  }
267 
268  CallbackData data;
269  data.length = file_length;
270 
271  if (!(data.infile = fopen(file_path.data(), "r"))) {
272  return Status(StrCat("Failed to open source file: ", file_path));
273  }
274 
275  S3ResponseHandler response_handler{S3File::PropertiesCallback, S3File::CompleteCallback};
276  S3PutObjectHandler obj_handler = { response_handler, &PutObjectDataCallback };
277 
278  for (unsigned attempts = 0; attempts < 5; ++attempts) {
279  S3_put_object(res.second, res.first.data(), file_length, nullptr, nullptr,
280  &obj_handler, &data);
281  if (data.status == S3StatusOK) {
282  break;
283  }
284  LOG(WARNING) << "Failed to put file to s3 " << file_name << ". Attempt " << attempts << " failed";
285  }
286 
287  CHECK(data.infile != 0);
288  if (fclose(data.infile) != 0) {
289  LOG(WARNING) << "Failed to close file: " << file_path;
290  }
291  if (data.status != S3StatusOK) {
292  return Status(StatusCode::IO_ERROR, S3_get_status_name(data.status));
293  }
294  return Status::OK;
295 }

328 static Status ListFilesInternal(StringPiece path, unsigned max_items, std::vector<StatShort>* file_names) {
329  if (!IsInS3Namespace(path)) {
330  return Status("Non-S3 path");
331  }
332 
333  auto res = GetKeyAndBucket(path);
334  const S3BucketContext* context = res.second;
335 
336  CallbackData callback_data;
337  callback_data.file_arr = file_names;
338  callback_data.bcontext = context;
339 
340  S3ResponseHandler response_handler{S3File::PropertiesCallback, S3File::CompleteCallback};
341  S3ListBucketHandler handler{response_handler, &MyListFileCb};
342 
343  string prefix = res.first.as_string();
344 
345  VLOG(1) << prefix << " " << max_items;
346  const char* marker = nullptr;
347  do {
348  for (unsigned attempts = 0; attempts < 5; ++attempts) {
349  S3_list_bucket(context, prefix.c_str(), marker, "/",
350  max_items, (S3RequestContext*)nullptr, &handler, &callback_data);
351  if (callback_data.status == S3StatusOK) {
352  break;
353  }
354  LOG(WARNING) << "Failed to list s3 bucket " << prefix << ". Attempt " << attempts << " failed";
355  }
356  if (callback_data.status != S3StatusOK) {
357  return Status(StatusCode::IO_ERROR, S3_get_status_name(callback_data.status));
358  }
359  marker = callback_data.marker.c_str();
360  } while (callback_data.is_truncated);
361 
362  return Status::OK;
363 }

365 Status ListFiles(StringPiece path, unsigned max_items, std::vector<StatShort>* file_names) {
366  auto pos = path.find('*', 0);
367  if (pos != std::string::npos) { // If path contains '*'.
368  StringPiece pattern = StringPiece(path, pos, path.length());
369  StringPiece path_without_star = StringPiece(path, 0, pos);
370  RETURN_IF_ERROR(ListFilesInternal(path_without_star, max_items, file_names));
371  // Remove files that do not match pattern.
372  auto out_it = file_names->begin();
373  auto pattern_str = pattern.as_string().c_str();
374  for (auto inp_it = file_names->begin(); inp_it != file_names->end();) {
375  if (0 != fnmatch(pattern_str, inp_it->name.c_str(), FNM_EXTMATCH)) {
376  ++inp_it;
377  } else {
378  *out_it++ = *inp_it++;
379  }
380  }
381  auto new_length = out_it - file_names->begin();
382  file_names->resize(new_length);
383  } else {
384  RETURN_IF_ERROR(ListFilesInternal(path, max_items, file_names));
385  }
386  return Status::OK;
387 }
