362 static void BM_LoopSingle(benchmark::State& state) {
363  int const min_exp = 0;
364  int const max_exp = (1 << 8) - 1; // exclusive!
365 
366  uint32_t bits = min_exp << 23;
367 
368  char str[32];
369 
370  while (state.KeepRunning()) {
371  float const f = IEEEFloat<float>(bits).value;
372  dtoa::ToString(f, str);
373 
374  ++bits;
375 
376  int next_exp = bits >> 23;
377  if (next_exp == max_exp)
378  break;
379  }
