16 /*
17  This is single producer - single consumer thread-safe channel.
18  It's fiber friendly, which means, multiple fibers at each end-point can use the channel:
19  K fibers from producer thread can push and N fibers from consumer thread can pull the records.
20  It has blocking interface that suspends blocked fibers upon empty/full conditions.
21  It also can be used for single-thread/multiple-fibers scenario.
22  This class designed to be pretty efficient by reducing the contention on its synchronization
23  primitives to minimum.
24 */
25 template <typename T> class SimpleChannel {
26  typedef ::boost::fibers::context::wait_queue_t wait_queue_t;
27  using spinlock_lock_t = ::boost::fibers::detail::spinlock_lock;
28 
29  public:
30  SimpleChannel(size_t n) : q_(n) {}
31 
32  template <typename... Args> void Push(Args&&... recordArgs) noexcept;
33 
34  // Blocking call. Returns false if channel is closed, true otherwise with the popped value.
35  bool Pop(T& dest);
36 
37  // Should be called only from the producer side. Signals the consumers that the channel
38  // is going to be close. Consumers may still pop the existing items until Pop() return false.
39 
40  // This function does not block, only puts the channel into closing state.
41  // It's responsibility of the caller to wait for the consumers to empty the remaining items
42  // and stop using the channel.
43  void StartClosing();
44 
45  // Non blocking
46  template <typename... Args> bool TryPush(Args&&... args) noexcept {
47  if (q_.write(std::forward<Args>(args)...)) {
48  if (++throttled_pushes_ > q_.capacity() / 3) {
49  pop_ec_.notify();
50  throttled_pushes_ = 0;
51  }
52  return true;
53  }
54  return false;
55  }
56 
57  bool TryPop(T& val) {
58  if (q_.read(val)) {
59  return true;
60  }
61  push_ec_.notify();
62  return false;
63  }
64 
65  bool IsClosing() const { return is_closing_.load(std::memory_order_relaxed); }
66 
67  private:
68  unsigned throttled_pushes_ = 0;
69 
70  folly::ProducerConsumerQueue<T> q_;
71  std::atomic_bool is_closing_{false};
72 
73  // Event counts provide almost negligible contention during fast-path (a single atomic add).
74  EventCount push_ec_, pop_ec_;
75 };

77 template <typename T>
78 template <typename... Args>
79 void SimpleChannel<T>::Push(Args&&... args) noexcept {
80  while (true) {
81  EventCount::Key key = push_ec_.prepareWait();
82  if (TryPush(std::forward<Args>(args)...)) {
83  break;
84  }
85  push_ec_.wait(key.epoch());
86  }
87 }

89 template <typename T> bool SimpleChannel<T>::Pop(T& dest) {
90  if (TryPop(dest))
91  return true;
92 
93  while (true) {
94  EventCount::Key key = pop_ec_.prepareWait();
95  if (TryPop(dest)) {
96  return true;
97  }
98 
99  if (is_closing_.load(std::memory_order_acquire)) {
100  return false;
101  }
102 
103  pop_ec_.wait(key.epoch());
104  }
105 }

107 template <typename T> void SimpleChannel<T>::StartClosing() {
108  // Full barrier, StartClosing performance does not matter.
109  is_closing_.store(true, std::memory_order_seq_cst);
110  pop_ec_.notifyAll();
111 }
