45 void BlockHeader::SetCrcAndLength(const uint8* ptr, size_t length) {
46  coding::EncodeFixed32(length, buf_ + 4);
47 
48  // Compute the crc of the record type and the payload.
49  uint32 crc = crc32c::Crc32c(buf_ + 8, 1);
50  crc = crc32c::Extend(crc, ptr, length);
51  crc = list_file::Mask(crc); // Adjust for storage
52  VLOG(2) << "EmitPhysicalRecord: type " << (buf_[8] & 0xF) << ", length: " << length
53  << ", crc: " << crc << " compressed: " << (buf_[8] & kCompressedMask);
54 
55  coding::EncodeFixed32(crc, buf_);
56 }

84 class FileHeader {
85  uint8 buf_[kListFileHeaderSize];
86  const std::map<string, string>& meta_;
87 
88  util::GrowableEncoder encoder_;
89 
90  public:
91  FileHeader(uint8 multiplier, const std::map<string, string>& meta) : meta_(meta) {
92  memcpy(buf_, kMagicString, kMagicStringSize);
93  buf_[kMagicStringSize] = multiplier;
94  buf_[kMagicStringSize + 1] = meta.empty() ? kNoExtension : kMetaExtension;
95  }

97  Status Write(util::Sink* sink) {
98  strings::ByteRange pc(buf_, sizeof(buf_));
99 
100  RETURN_IF_ERROR(sink->Append(pc));
101  if (!meta_.empty()) {
102  // Meta format: crc32, fixed32 - meta block size, varint32 map size,
103  // (varint string size, string data)+
104  // We do not bother with memory optimizations since the meta data should relatively small.
105  // We do not bother with memory optimizations since the meta data should relatively small.
106  encoder_.put_varint32(meta_.size());
107  for (const auto& k_v : meta_) {
108  encoder_.put_varint32(k_v.first.size());
109  encoder_.put_string(k_v.first);
110  encoder_.put_varint32(k_v.second.size());
111  encoder_.put_string(k_v.second);
112  }
113 
114  uint8 meta_header[8];
115  coding::EncodeFixed32(encoder_.size(), meta_header + 4);
116  uint32 crc = list_file::Mask(crc32c::Crc32c(encoder_.byteptr(), encoder_.size()));
117  coding::EncodeFixed32(crc, meta_header);
118 
119  RETURN_IF_ERROR(sink->Append(ByteRange(meta_header, sizeof meta_header)));
120  RETURN_IF_ERROR(sink->Append(ByteRange{encoder_.byteptr(), encoder_.size()}));
121  }
122  return Status::OK;
123  }

203 inline Status Lst1Impl::FlushArray() {
204  if (array_records_ == 0)
205  return Status::OK;
206 
207  Varint32Encoder enc(array_records_);
208 
209  // We prepend array_records_ integer right before the data, for that we skip
210  // kArrayRecordMaxHeaderSize - enc.size() bytes.
211  uint8* start = array_store_.get() + kArrayRecordMaxHeaderSize - enc.size();
212  memcpy(start, enc.data(), enc.size());
213 
214  // Flush the array.
215  Status st = EmitPhysicalRecord(kArrayType, start, array_next_ - start);
216  array_records_ = 0;
217  return st;

220 Status Lst1Impl::AddRecord(StringPiece record) {
221  CHECK_GT(block_size_, 0) << "ListWriter::Init was not called.";
222 
223  Varint32Encoder record_size_encoded(record.size());
224  const uint32 record_size_total = record_size_encoded.size() + record.size();
225  // Try to accomodate either in the array or a single block. Multiple iterations might be
226  // needed since we might fragment the record.
227  bool fragmenting = false;
228  ++records_added_;
229  while (true) {
230  if (array_records_ > 0) {
231  if (array_next_ + record_size_total <= array_end_) {
232  AddRecordToArray(record_size_encoded.slice(), record);
233  return Status::OK;
234  }
235  RETURN_IF_ERROR(FlushArray());
236  // Also we must either split the record or transfer to the next block.
237  }
238  if (block_leftover() <= kBlockHeaderSize) {
239  // Block trailing bytes. Just fill them with zeroes.
240  uint8 kBlockFilling[kBlockHeaderSize] = {0};
241  RETURN_IF_ERROR(dest_->Append(ByteRange(kBlockFilling, block_leftover())));
242  block_offset_ = 0;
243  block_leftover_ = block_size_;
244  }
245 
246  if (fragmenting) {
247  size_t fragment_length = record.size();
248  RecordType type = kLastType;
249  if (fragment_length > block_leftover() - kBlockHeaderSize) {
250  fragment_length = block_leftover() - kBlockHeaderSize;
251  type = kMiddleType;
252  }
253  RETURN_IF_ERROR(EmitPhysicalRecord(type, u8ptr(record.data()), fragment_length));
254  if (type == kLastType)
255  return Status::OK;
256  record.remove_prefix(fragment_length);
257  continue;
258  }
259  if (record_size_total + kArrayRecordMaxHeaderSize < block_leftover()) {
260  // Lets start the array accumulation.
261  // We leave space at the beginning to prepend the header at the end.
262  array_next_ = array_store_.get() + kArrayRecordMaxHeaderSize;
263  array_end_ = array_store_.get() + block_leftover();
264  AddRecordToArray(record_size_encoded.slice(), record);
265  return Status::OK;
266  }
267  if (kBlockHeaderSize + record.size() <= block_leftover()) {
268  // We have space for one record in this block but not for the array.
269  return EmitPhysicalRecord(kFullType, u8ptr(record), record.size());
270  }
271  // We must fragment.
272  fragmenting = true;
273  const size_t fragment_length = block_leftover() - kBlockHeaderSize;
274  RETURN_IF_ERROR(EmitPhysicalRecord(kFirstType, u8ptr(record), fragment_length));
275  record.remove_prefix(fragment_length);
276  };
277  return Status(StatusCode::INTERNAL_ERROR, "Should not reach here");
278 }

282 Status Lst1Impl::EmitPhysicalRecord(RecordType type, const uint8* ptr, size_t length) {
283  DCHECK_LE(kBlockHeaderSize + length, block_leftover());
284 
285  // Format the header
286  BlockHeader block_header(type);
287 
288  if (options_.use_compression && length > 64) {
289  size_t compressed_length = compress_buf_size_;
290  auto status = compress_func_(options_.compress_level, ptr, length, compress_buf_.get() + 1,
291  &compressed_length);
292  if (status.ok()) {
293  VLOG(1) << "Compressed record with size " << length << " to ratio "
294  << float(compressed_length) / length;
295  if (compressed_length < length - length / kCompressReduction) {
296  block_header.EnableCompression();
297 
298  compress_buf_[0] = options_.compress_method;
299  ptr = compress_buf_.get();
300  compression_savings_ += (length - compressed_length);
301  length = compressed_length + 1;
302  }
303  }
304  }
305  block_header.SetCrcAndLength(ptr, length);
306 
307  // Write the header and the payload
308  RETURN_IF_ERROR(block_header.Write(dest_.get()));
309  RETURN_IF_ERROR(dest_->Append(strings::ByteRange(ptr, length)));
310 
311  bytes_added_ += (kBlockHeaderSize + length);
312  block_offset_ += (kBlockHeaderSize + length);
313  block_leftover_ = block_size_ - block_offset_;
314  return Status::OK;
315 }

318 ListWriter::ListWriter(StringPiece filename, const Options& options) {
319  Options opts = options;
320  size_t header_offset = 0;
321  size_t file_offset = 0;
322 
323  if (options.append) {
324  auto status_obj = ReadonlyFile::Open(filename);
325  if (status_obj.ok()) {
326  HeaderParser parser;
327  std::map<string, string> meta;
328  if (parser.Parse(status_obj.obj, &meta).ok()) {
329  opts.block_size_multiplier = parser.block_multiplier();
330  header_offset = parser.offset();
331  file_offset = status_obj.obj->Size();
332 
333  CHECK_GE(file_offset, header_offset);
334  }
335  status_obj.obj->Close();
336  delete status_obj.obj;
337  }
338  }
339 
340  opts.append = header_offset > 0;
341  opts.internal_append_offset = file_offset - header_offset;
342  file::OpenOptions open_options;
343  open_options.append = opts.append;
344  WriteFile* file = file::Open(filename, open_options);
345 
346  impl_.reset(new Lst1Impl(new Sink(file, TAKE_OWNERSHIP), opts));
347 }
