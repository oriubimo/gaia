32 static void HandleCpuProfile(bool enable, StringResponse* response) {
33  string profile_name = "/tmp/" + base::ProgramBaseName();
34  response->set(h2::field::cache_control, "no-cache, no-store, must-revalidate");
35  response->set(h2::field::pragma, "no-cache");
36  response->set(field::content_type, kHtmlMime);
37 
38  auto& body = response->body();
39 
40  if (enable) {
41  if (last_profile_suffix[0]) {
42  body.append("<p> Yo, already profiling, stupid!</p>\n");
43  } else {
44  string suffix = base::LocalTimeNow("_%d%m%Y_%H%M%S.prof");
45  profile_name.append(suffix);
46  strcpy(last_profile_suffix, suffix.c_str());
47  int res = ProfilerStart(profile_name.c_str());
48  LOG(INFO) << "Starting profiling into " << profile_name << " " << res;
49  body.append("<p> Yeah, let's profile this bitch, baby!</p> \n"
50  "<img src='https://gist.githack.com/romange/4760c3eebc407755f856fec8e5b6d4c1/raw/profiler.gif'>\n");
51  }
52  return;
53  }
54  ProfilerStop();
55  if (last_profile_suffix[0] == '\0') {
56  body.append("<h3>Profiling is off, commander!</h3> \n");
57  return;
58  }
59  string cmd("nice -n 15 pprof -noinlines -lines -unit ms --svg ");
60  string symbols_name = base::ProgramAbsoluteFileName() + ".debug";
61  LOG(INFO) << "Symbols " << symbols_name << ", suffix: " << last_profile_suffix;
62  if (access(symbols_name.c_str(), R_OK) != 0) {
63  symbols_name = base::ProgramAbsoluteFileName();
64  }
65  cmd.append(symbols_name).append(" ");
66 
67  profile_name.append(last_profile_suffix);
68  cmd.append(profile_name).append(" > ");
69 
70  string err_log = profile_name + ".err";
71  profile_name.append(".svg");
72 
73  cmd.append(profile_name).append(" 2> ").append(err_log);
74 
75  LOG(INFO) << "Running command: " << cmd;
76  last_profile_suffix[0] = '\0';
77 
78  int sh_res = util::sh_exec(cmd.c_str());
79  if (sh_res != 0) {
80  LOG(ERROR) << "Error running sh_exec, status: " << errno << " " << strerror(errno);
81  }
82 
83  // Redirect browser to show this file.
84  string url("filez?file=");
85  url.append(profile_name);
86  LOG(INFO) << "Redirecting to " << url;
87  google::FlushLogFiles(google::INFO);
88 
89  response->set(h2::field::location, url);
90  response->result(h2::status::moved_permanently);
91 }

94 void ProfilezHandler(const QueryArgs& args, HttpHandler::SendFunction* send) {
95  bool enable = false;
96  for (const auto& k_v : args) {
97  if (k_v.first == "profile" && k_v.second == "on")
98  enable = true;
99  }
100 
101  fibers_ext::Done done;
102  std::thread([=] () mutable {
103  StringResponse response;
104 
105  HandleCpuProfile(enable, &response);
106  send->Invoke(std::move(response));
107  done.Notify();
108  }).detach();
109 
110  // Instead of joining the thread which is not fiber-friendly,
111  // I use done to block the fiber but free the thread to handle other fibers.
112  // Still this fiber connection is blocked.
113  done.Wait();
114 }
