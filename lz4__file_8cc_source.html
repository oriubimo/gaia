24 bool LZ4File::Open() {
25  CHECK_EQ(0, fd_);
26  int flags = O_CREAT | O_WRONLY | O_CLOEXEC | O_TRUNC;
27  fd_ = open(create_file_name_.c_str(), flags, 0666);
28  if (fd_ < 0) {
29  LOG(ERROR) << "Could not open file " << strerror(errno);
30  return false;
31  }
32 
33  LZ4F_preferences_t prefs;
34  LZ4F_errorCode_t result = LZ4F_createCompressionContext(&context_, LZ4F_VERSION);
35  CHECK(!LZ4F_isError(result)) << LZ4F_getErrorName(result);
36 
37  /* Init */
38  memset(&prefs, 0, sizeof(prefs));
39 
40  prefs.autoFlush = 1;
41  prefs.compressionLevel = level_;
42  prefs.frameInfo.blockSizeID = LZ4F_max64KB;
43  char dst_buf[32];
44 
45  size_t header_size = LZ4F_compressBegin(context_, dst_buf, sizeof(dst_buf), &prefs);
46  CHECK(!LZ4F_isError(header_size)) << LZ4F_getErrorName(header_size);
47 
48  ssize_t sz = write(fd_, dst_buf, header_size);
49  if (sz < 0) {
50  LOG(ERROR) << "Could not open file " << strerror(errno);
51  LZ4F_freeCompressionContext(context_);
52  close(fd_);
53  fd_ = 0;
54  return false;
55  }
56 
57  CHECK_EQ(header_size, sz);
58  buf_size_ = LZ4F_compressBound(kInputBatchSize, &prefs);
59 
60  buf_ = new uint8[buf_size_];
61  return true;
62 }

64 bool LZ4File::Close() {

66 
67  size_t header_size = LZ4F_compressEnd(context_, buf_, buf_size_, NULL);
68  if (LZ4F_isError(header_size)) {
69  LOG(ERROR) << "Error closing file " << LZ4F_getErrorName(header_size);
70  return false;
71  }
72 
73  ssize_t sz = write(fd_, buf_, header_size);
74  if (sz < 0) {
75  LOG(ERROR) << "Error closing file " << strerror(errno);
76  return false;
77  }
78  close(fd_);
79  LZ4F_freeCompressionContext(context_);
80 
81  delete this;
82 
83  return true;
84 }

86 Status LZ4File::Write(const uint8* buffer, uint64 length) {
87  do {
88  size_t bl = std::min(length, kInputBatchSize);
89  size_t written = LZ4F_compressUpdate(context_, buf_, buf_size_, buffer, bl, NULL);
90  if (LZ4F_isError(written)) {
91  return Status(LZ4F_getErrorName(written));
92  }
93  buffer += bl;
94  length -= bl;
95  if (written) {
96  ssize_t sz = write(fd_, buf_, written);
97  if (sz < 0) {
98  return StatusFileError();
99  }
100  }
101 
102  } while (length);
103  return Status::OK;
104 }
