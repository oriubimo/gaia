65  void property_change(boost::fibers::context* ctx, IoFiberProperties& props) noexcept final {
66  // Although our priority_props class defines multiple properties, only
67  // one of them (priority) actually calls notify() when changed. The
68  // point of a property_change() override is to reshuffle the ready
69  // queue according to the updated priority value.
70 
71  // 'ctx' might not be in our queue at all, if caller is changing the
72  // priority of (say) the running fiber. If it's not there, no need to
73  // move it: we'll handle it next time it hits awakened().
74  if (!ctx->ready_is_linked()) {
75  return;
76  }
77 
78  // Found ctx: unlink it
79  ctx->ready_unlink();
80  if (!ctx->is_context(fibers::type::dispatcher_context)) {
81  DCHECK_GT(ready_cnt_, 0);
82  --ready_cnt_;
83  }
84 
85  // Here we know that ctx was in our ready queue, but we've unlinked
86  // it. We happen to have a method that will (re-)add a context* to the
87  // right place in the ready queue.
88  awakened(ctx, props);
89  }

93  // suspend_until halts the thread in case there are no active fibers to run on it.
94  // This is done by dispatcher fiber.
95  void suspend_until(chrono::steady_clock::time_point const& abs_time) noexcept final {
96  DVLOG(2) << "suspend_until " << abs_time.time_since_epoch().count();
97 
98  // Only dispatcher context stops the thread.
99  // Which means that either the main loop in MAIN_LOOP_SUSPEND state or it finished and we
100  // are in shutdown phase.
101  // suspend_until is responsible for THREAD SUSPEND which is not what MAIN_LOOP_SUSPEND for.
102  // In fact, we implement THREAD SUSPEND by deferring the control to asio::io_context::run_one.
103  // For that to happen we need to resume the MAIN_LOOP_SUSPEND, and that's why we
104  // schedule main_loop_ctx_ below.
105  DCHECK(fibers::context::active()->is_context(fibers::type::dispatcher_context));
106 
107  // Set a timer so at least one handler will eventually fire, causing
108  // run_one() to eventually return.
109  if (abs_time != STEADY_PT_MAX && suspend_tp_ != abs_time) {
110  // Each expires_at(time_point) call cancels any previous pending
111  // call. We could inadvertently spin like this:
112  // dispatcher calls suspend_until() with earliest wake time
113  // suspend_until() sets suspend_timer_,
114  // loop() calls run_one()
115  // some other asio handler runs before timer expires
116  // run_one() returns to loop()
117  // loop() yields to dispatcher
118  // dispatcher finds no ready fibers
119  // dispatcher calls suspend_until() with SAME wake time
120  // suspend_until() sets suspend_timer_ to same time, canceling
121  // previous async_wait()
122  // loop() calls run_one()
123  // asio calls suspend_timer_ handler with operation_aborted
124  // run_one() returns to loop()... etc. etc.
125  // So only actually set the timer when we're passed a DIFFERENT
126  // abs_time value.
127  suspend_tp_ = abs_time;
128  suspend_timer_->expires_at(abs_time);
129 
130  suspend_timer_->async_wait([this, abs_time](const system::error_code& ec) {
131  // Call Suspend handler.
132  SuspendCb(ec, abs_time);
133  });
134  VLOG(1) << "Arm suspender at micros from now " << delta_micros(abs_time)
135  << ", abstime: " << abs_time.time_since_epoch().count();
136  }
137  CHECK_EQ(0, mask_ & LOOP_RUN_ONE) << "Deadlock detected";
138 
139  // Awake main_loop_ctx_ in WaitTillFibersSuspend().
140  main_loop_ctx_->get_scheduler()->schedule(main_loop_ctx_);
141  }
142  //]
143 
144  // This function is called from remote threads, to wake this thread in case it's sleeping.
145  // In our case, "sleeping" means - might stuck inside run_one waiting for events.
146  // We break from run_one() by scheduling a timer.
147  void notify() noexcept final;
148 
149  void MainLoop();
150 
151  private:
152  void SuspendCb(const system::error_code& ec, chrono::steady_clock::time_point tp) {
153  VLOG(1) << "Fire suspender " << tp.time_since_epoch().count() << " " << ec;
154  if (!ec) {
155  // The timer has successfully finished.
156  suspend_tp_ = STEADY_PT_MAX;
157 
158  // Switch to dispatch fiber to awaken fibers.
159  this_fiber::yield();
160  } else {
161  CHECK_EQ(ec, asio::error::operation_aborted);
162  }
163  }
164  void WaitTillFibersSuspend();
165 };

169 void AsioScheduler::MainLoop() {
170  asio::io_context* io_cntx = io_context_.get();
171  main_loop_ctx_ = fibers::context::active();
172 
173  while (!io_cntx->stopped()) {
174  if (has_ready_fibers()) {
175  while (io_cntx->poll())
176  ;
177 
178  // Gives up control to allow other fibers to run in the thread.
179  WaitTillFibersSuspend();
180  continue;
181  }
182 
183  // run one handler inside io_context
184  // if no handler available, blocks this thread
185  DVLOG(2) << "MainLoop::RunOneStart";
186  mask_ |= LOOP_RUN_ONE;
187  if (!io_cntx->run_one()) {
188  mask_ &= ~LOOP_RUN_ONE;
189  break;
190  }
191  DVLOG(2) << "MainLoop::RunOneEnd";
192  mask_ &= ~LOOP_RUN_ONE;
193  }
194 
195  VLOG(1) << "MainLoop exited";
196  mask_ |= MAIN_LOOP_FINISHED;
197 
198  // We won't run "run_one" anymore therefore we can remove our dependence on suspend_timer_ and
199  // asio:context. In fact, we can not use timed-waits from now on using fibers code because we've
200  // relied on asio for that.
201  // We guard suspend_timer_ using notify_guard_ with no locks in notify().
202  // However we must block here.
203  // It's a shutdown phase, so we do not care. We can not rely on fiber scheduling methods since
204  // they rely on the loop above. Therefore we just use thread yield.
205 
206  // Signal that we shutdown and check if we need to wait for current notify calls to exit.
207  constexpr uint32_t NOTIFY_BIT = 1U << NOTIFY_GUARD_SHIFT;
208  uint32_t seq = notify_guard_.fetch_or(NOTIFY_BIT);
209  while (seq) {
210  pthread_yield(); // almost like a sleep
211  seq = notify_guard_.load() & (NOTIFY_BIT - 1); // Block untill all notify calls exited.
212  }
213  suspend_timer_.reset(); // now we can free suspend_timer_.
214 
215  VLOG(1) << "MainLoopWakes/NotifyCnt: " << main_loop_wakes_ << "/" << notify_cnt_;
216 }

218 void AsioScheduler::WaitTillFibersSuspend() {
219  // block this fiber till all (ready) fibers are processed
220  // or when AsioScheduler::suspend_until() has been called or awaken() decided to resume it.
221  mask_ |= MAIN_LOOP_SUSPEND;
222 
223  DVLOG(2) << "WaitTillFibersSuspend:Start";
224  main_loop_ctx_->suspend();
225  mask_ &= ~MAIN_LOOP_SUSPEND;
226  switch_cnt_ = 0;
227  DVLOG(2) << "WaitTillFibersSuspend:End";
228 }

231 void AsioScheduler::awakened(fibers::context* ctx, IoFiberProperties& props) noexcept {
232  DCHECK(!ctx->ready_is_linked());
233 
234  ready_queue_type* rq;
235  if (ctx->is_context(fibers::type::dispatcher_context)) {
236  rq = rqueue_arr_ + DISPATCH_LEVEL;
237  DVLOG(2) << "Ready: " << fibers_ext::short_id(ctx) << " dispatch"
238  << ", ready_cnt: " << ready_cnt_;
239  } else {
240  unsigned nice = props.nice_level();
241  DCHECK_LT(nice, IoFiberProperties::NUM_NICE_LEVELS);
242  rq = rqueue_arr_ + nice;
243  ++ready_cnt_;
244  if (last_nice_level_ > nice)
245  last_nice_level_ = nice;
246 
247  // In addition, we wake main_loop_ctx_ is too many switches ocurred
248  // while it was suspended.
249  // It's a convenient place to wake because we are sure there is a least
250  // one ready worker in addition to main_loop_ctx_ and it won't stuck in
251  // run_one().
252  // * main_loop_ctx_->ready_is_linked() could be linked already in the previous invocations
253  // of awakened before pick_next resumed it.
254  if (nice > MAIN_NICE_LEVEL && switch_cnt_ > MAIN_SWITCH_LIMIT &&
255  !main_loop_ctx_->ready_is_linked()) {
256  DVLOG(2) << "Wake MAIN_LOOP_SUSPEND " << fibers_ext::short_id(main_loop_ctx_)
257  << ", r/s: " << ready_cnt_ << "/" << switch_cnt_;
258 
259  switch_cnt_ = 0;
260  ++ready_cnt_;
261  main_loop_ctx_->ready_link(rqueue_arr_[MAIN_NICE_LEVEL]);
262  last_nice_level_ = MAIN_NICE_LEVEL;
263  ++main_loop_wakes_;
264  }
265 
266  DVLOG(2) << "Ready: " << fibers_ext::short_id(ctx) << "/" << props.name()
267  << ", nice/rdc: " << nice << "/" << ready_cnt_;
268  }
269 
270  ctx->ready_link(*rq); /*< fiber, enqueue on ready queue >*/
271 }

273 fibers::context* AsioScheduler::pick_next() noexcept {
274  fibers::context* ctx(nullptr);
275  using fibers_ext::short_id;
276 
277  for (; last_nice_level_ < IoFiberProperties::NUM_NICE_LEVELS; ++last_nice_level_) {
278  auto& q = rqueue_arr_[last_nice_level_];
279  if (q.empty())
280  continue;
281 
282  // pop an item from the ready queue
283  ctx = &q.front();
284  q.pop_front();
285 
286  DCHECK(!ctx->is_context(fibers::type::dispatcher_context));
287  DCHECK_GT(ready_cnt_, 0);
288  --ready_cnt_;
289 
290  DVLOG(2) << "Switching from " << short_id() << " to " << short_id(ctx) << " switch_cnt("
291  << switch_cnt_ << ")";
292  DCHECK(ctx != fibers::context::active());
293 
294  // Checking if we want to resume to main loop prematurely to preserve responsiveness
295  // of IO loop. MAIN_NICE_LEVEL is reserved for the main loop so we count only
296  // when we switch to other fibers and the loop is in MAIN_LOOP_SUSPEND state.
297  if ((mask_ & MAIN_LOOP_SUSPEND) && last_nice_level_ > MAIN_NICE_LEVEL) {
298  ++switch_cnt_;
299 
300  // We can not prematurely wake MAIN_LOOP_SUSPEND if ready_cnt_ == 0.
301  // The reason for this is that in that case the main loop will call "run_one" during the
302  // next iteration and might get stuck there because we never reached suspend_until
303  // that might configure asio loop to break early.
304  // This is why we break from MAIN_LOOP_SUSPEND inside waken call, where
305  // we are sure there is at least one worker fiber, and the main loop won't stuck in run_one.
306  }
307 
308  DVLOG(3) << "pick_next: " << short_id(ctx);
309 
310  return ctx;
311  }
312 
313  DCHECK_EQ(0, ready_cnt_);
314 
315  auto& dispatch_q = rqueue_arr_[DISPATCH_LEVEL];
316  if (!dispatch_q.empty()) {
317  fibers::context* ctx = &dispatch_q.front();
318  dispatch_q.pop_front();
319 
320  DVLOG(2) << "Switching from " << short_id() << " to dispatch " << short_id(ctx)
321  << ", mask: " << unsigned(mask_);
322  return ctx;
323  }
324 
325  DVLOG(2) << "pick_next: null";
326 
327  return nullptr;
328 }

331  uint32_t seq = notify_guard_.fetch_add(1, std::memory_order_acq_rel);
332 
333  if ((seq >> 16) == 0) { // Test whether are not shuttind down.
334 
335  // We need to break from run_one asap.
336  // If we've already armed suspend_timer_ via suspend_until then cancelling it would suffice.
337  // However, in case suspend_until has not been called or called STEADY_PT_MAX,

339  // In addition, AsioScheduler::notify is called from remote thread so may only access
340  // thread-safe data. Therefore, the simplest solution would be just post a callback.
341  VLOG(1) << "AsioScheduler::notify";
342 
343  /*suspend_timer_->expires_at(chrono::steady_clock::now());
344  suspend_timer_->async_wait([this](const system::error_code& ec) {

371 void IoContext::StartLoop(BlockingCounter* bc) {
372  // I do not use use_scheduling_algorithm because I want to retain access to the scheduler.
373  // fibers::use_scheduling_algorithm<AsioScheduler>(io_ptr);
374  AsioScheduler* scheduler = new AsioScheduler(context_ptr_);
375  fibers::context::active()->get_scheduler()->set_algo(scheduler);
376  this_fiber::properties<IoFiberProperties>().set_name("io_loop");
377  this_fiber::properties<IoFiberProperties>().SetNiceLevel(MAIN_NICE_LEVEL);
378  CHECK(fibers::context::active()->is_context(fibers::type::main_context));
379 
380  thread_id_ = this_thread::get_id();
381 
382  io_context& io_cntx = *context_ptr_;
383 
384  // We run the main loop inside the callback of io_context, blocking it until the loop exits.
385  // The reason for this is that io_context::running_in_this_thread() is deduced based on the
386  // call-stack. GAIA code should use InContextThread() to check whether the code runs in the
387  // context's thread.
388  Async([scheduler, bc] {
389  bc->Dec();
390  scheduler->MainLoop();
391  });
392 
393  // Bootstrap - launch the callback handler above.
394  // It will block until MainLoop exits. See comment above.
395  io_cntx.run_one();
396 
397  // Shutdown phase.
398  for (unsigned i = 0; i < 2; ++i) {
399  DVLOG(1) << "Cleanup Loop " << i;
400  while (io_cntx.poll() || scheduler->has_ready_fibers()) {
401  this_fiber::yield(); // while something happens, pass the ownership to other fiber.
402  }
403  io_cntx.restart();
404  }
405 }

407 void IoContext::Stop() {
408  if (cancellable_arr_.size() > 0) {
409  fibers_ext::BlockingCounter cancel_bc(cancellable_arr_.size());
410 
411  VLOG(1) << "Cancelling " << cancellable_arr_.size() << " cancellables";
412  // Shutdown sequence and cleanup.
413  for (auto& k_v : cancellable_arr_) {
414  AsyncFiber([&] {
415  k_v.first->Cancel();
416  cancel_bc.Dec();
417  });
418  }
419  cancel_bc.Wait();
420  for (auto& k_v : cancellable_arr_) {
421  k_v.second.join();
422  }
423  cancellable_arr_.clear();
424  }
425 
426  context_ptr_->stop();
427  VLOG(1) << "AsioIoContext stopped";
428 }
