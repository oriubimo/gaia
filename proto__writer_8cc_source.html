73 util::Status ListProtoWriter::Add(const ::google::protobuf::MessageLite& msg) {
74  CHECK(writer_);
75  CHECK_EQ(dscr_->full_name(), msg.GetTypeName());
76  if (!was_init_) {
77  RETURN_IF_ERROR(writer_->Init());

81  if (options_.max_entries_per_file > 0 && ++entries_per_shard_ > options_.max_entries_per_file) {
82  RETURN_IF_ERROR(writer_->Flush());

110 string GenerateSerializedFdSet(const gpb::Descriptor* dscr) {
111  const gpb::FileDescriptor* fd = dscr->file();
112  gpb::FileDescriptorSet fd_set;
113  std::unordered_set<const gpb::FileDescriptor*> unique_set({fd});
114  std::vector<const gpb::FileDescriptor*> stack({fd});
115  while (!stack.empty()) {
116  fd = stack.back();
117  stack.pop_back();
118  fd->CopyTo(fd_set.add_file());
119 
120  for (int i = 0; i < fd->dependency_count(); ++i) {
121  const gpb::FileDescriptor* child = fd->dependency(i);
122  if (unique_set.insert(child).second) {
123  stack.push_back(child);
124  }
125  }
126  }
127  return fd_set.SerializeAsString();
128 }
