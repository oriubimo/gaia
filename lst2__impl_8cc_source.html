27 /* Block - divides files into equal parts of size BlockSize. Each Block size can be
28  in range :64k-16MB.
29  Each block consists of one or more records.
30 */
31 uint8_t* RecordHeader::Write(uint8_t* dest) const {
32  *dest++ = flags;
33  if (compress_method != list_file::kCompressionNone) {
34  *dest++ = compress_method;
35  }
36  if (arr_count) {
37  LittleEndian::Store16(dest, arr_count);
38  dest += sizeof(uint16_t);
39  }
40  LittleEndian::Store32(dest, size);
41  dest += (2 + (size > 0xFFFF));
42  LittleEndian::Store32(dest, crc);
43  dest += sizeof(uint32_t);
44 
45  return dest;
46 }

48 size_t RecordHeader::Parse(const uint8_t* src) {
49  flags = src[0];
50  const uint8_t* next = src + 1;
51 
52  if ((flags & kTypeMask) == kArrayType) {
53  arr_count = LittleEndian::Load16(next);
54  next += sizeof(uint16_t);
55  }
56  size = LittleEndian::Load16(next);
57  next += sizeof(uint16_t);
58 
59  if (flags & kRecordSize3BytesFlag) {
60  size |= (uint32_t(next[0]) << 16);
61  ++next;
62  }
63 
64  if ((flags & kCompressedFlag) == 0) {
65  crc = LittleEndian::Load32(next);
66  next += sizeof(uint32_t);
67  }
68 
69  return next - src;
70 }

121 /* Record Header:
122  |Type_Flags(1) | CompressType(1)? | ArrayCount(2)? | RecordSize (2-3) |CRC(4)? |
123  Followed by payload of length RecordSize.
124  For compressed payload there is no need fror CRC check.
125  Payload: (uint16_t item_sz[ArrayCount]) + ItemDataSize. // item_sz if item type is array.
126  Another option: To store varint_payload_sz + varint_sz_arr + ItemDataSize
127 */
128 Status Lst2Impl::AddRecord(StringPiece record) {
129  CHECK(init_called_) << "ListWriter::Init was not called.";
130 
131  if (record.size() >= (1ULL << 30)) {
132  return Status("Record too large");
133  }
134 
135  if (array_records_ > 0) {
136  if (record.size() + array_next_ < array_end_) {
137  uint8_t* next = Varint::Encode32(array_next_, record.size());
138  if (next + record.size() <= array_end_) {
139  memcpy(next, record.data(), record.size());
140  array_next_ = next + record.size();
141  ++array_records_;
142  if (array_records_ == kuint16max) {
143  RETURN_IF_ERROR(FlushArray());
144  }
145  return Status::OK;
146  }
147  }
148  RETURN_IF_ERROR(FlushArray());
149  }
150  DCHECK_EQ(0, array_records_);
151 
152  uint32_t block_left = block_size() - block_offset_;
153  DCHECK_GT(block_left, RecordHeader::kSingleSmallSize);
154 
155  auto wrapped_size = RecordHeader::WrappedSize(record.size());
156 
157  if (wrapped_size + RecordHeader::kArrayMargin <= block_left) {
158  array_next_ = array_store_.get();
159  array_end_ = array_store_.get() + block_left - RecordHeader::kMaxSize;
160  array_next_ = Varint::Encode32(array_next_, record.size());
161  memcpy(array_next_, record.data(), record.size());
162  array_next_ += record.size();
163  array_records_ = 1;
164  return Status::OK;
165  }
166 
167  if (wrapped_size <= block_left) {
168  // We have space for one record in this block but not for the array.
169  return EmitSingleRecord(kFullType, record);
170  }
171  return WriteFragmented(record);
172 }

176 Status Lst2Impl::FlushArray() {
177  if (array_records_ == 0)
178  return Status::OK;
179 
180  RecordHeader rh;
181  uint8_t buf[RecordHeader::kMaxSize];
182  rh.flags = kArrayType;
183  rh.arr_count = array_records_;
184  rh.size = array_end_ - array_store_.get();
185  rh.crc = crc32c::Crc32c(array_store_.get(), rh.size);
186 
187  uint8_t* next = rh.Write(buf);
188  uint32_t hs = next - buf;
189  strings::ByteRange rec(array_store_.get(), rh.size);
190 
191  // Flush the array.
192  Status st = EmitPhysicalRecord(strings::ByteRange(buf, hs), rec);
193  array_records_ = 0;
194 
195  return st;
196 }

198 util::Status Lst2Impl::WriteFragmented(StringPiece record) {
199  RecordType t = kFirstType;
200  RecordHeader rh;
201  uint8_t buf[RecordHeader::kMaxSize];
202 
203  uint32_t block_left = block_size() - block_offset_;
204  DCHECK_GT(block_left, RecordHeader::kSingleSmallSize);
205 
206  rh.size = RecordHeader::PayloadSize(block_left);
207  rh.flags = t;
208 
209  DCHECK(!options_.use_compression || list_file::kCompressionNone == options_.compress_method);
210 
211  while (true) {
212  rh.crc = crc32c::Crc32c(record.data(), rh.size);
213  uint8_t* next = rh.Write(buf);
214  strings::ByteRange header(buf, next - buf);
215  strings::ByteRange payload(strings::u8ptr(record.data()), rh.size);
216  RETURN_IF_ERROR(EmitPhysicalRecord(header, payload));
217  record.remove_prefix(rh.size);
218  if (record.empty())
219  break;
220 
221  DCHECK_EQ(0, block_offset_);
222  rh.size = std::min<uint32_t>(RecordHeader::PayloadSize(block_size_), record.size());
223  t = rh.size == record.size() ? kLastType : kMiddleType;
224  }
225  return Status::OK;
226 }

241 util::Status Lst2Impl::EmitPhysicalRecord(strings::ByteRange header, strings::ByteRange record) {
242  DCHECK_LE(block_offset_ + header.size() + record.size(), block_size_);
243 
244  RETURN_IF_ERROR(dest_->Append(header));
245  RETURN_IF_ERROR(dest_->Append(record));
246 
247  block_offset_ += (header.size() + record.size());
248 
249  if (block_offset_ + RecordHeader::kSingleSmallSize >= block_size_) {
250  if (block_offset_ < block_size_) {
251  uint8_t buf[RecordHeader::kSingleSmallSize] = {0};
252  RETURN_IF_ERROR(dest_->Append(strings::ByteRange(buf, block_size_ - block_offset_)));
253  }
254  block_offset_ = 0;
255  }
256  return Status::OK;

259 bool ReaderImpl::ReadHeader(std::map<std::string, std::string>* dest) {
260  uint8_t buf[4];
261  auto res = wrapper_->file->Read(kMagicStringSize, strings::MutableByteRange(buf, sizeof(buf)));
262  if (!res.ok())
263  return false;
264 
265  uint16_t multiplier = LittleEndian::Load16(buf);
266  uint16_t num_pairs = LittleEndian::Load16(buf + 2);
267 
268  if (multiplier == 0 || multiplier >= 256 || (num_pairs % 2) != 0) {
269  return false;
270  }
271  wrapper_->block_size = multiplier * kBlockSizeFactor;
272  file_offset_ = kListFileHeaderSize;
273 
274  if (kListFileHeaderSize >= wrapper_->file->Size()) {
275  wrapper_->eof = true;
276  }
277 
278  string key, val;
279 
280  for (unsigned i = 0; i < num_pairs; ++i) {
281  StringPiece skey, sval;
282 
283  if (!ReadRecord(&skey, &key))
284  return false;
285  if (!ReadRecord(&sval, &val))
286  return false;
287  dest->emplace(string(skey), string(sval));
288  }
289 
290  // We allocate more to allow simpler parsing.
291  backing_store_.reset(new uint8_t[wrapper_->block_size + 8]);
292  return true;
293 }

295 bool ReaderImpl::ReadRecord(StringPiece* record, std::string* scratch) {
296  bool in_fragmented_record = false;
297  scratch->clear();
298 
299  while (true) {
300  if (!array_store_.empty()) {
301  uint32_t len;
302  const uint8_t* start = strings::u8ptr(array_store_.data());
303  const uint8_t* next = Varint::Parse32Inline(start, &len);
304  size_t s = next - start + len;
305  if (s > array_store_.size()) {
306  wrapper_->ReportCorruption(array_store_.size(), "Invalid array record");
307  array_store_ = StringPiece();
308  continue;
309  }
310 
311  *record = StringPiece(strings::charptr(next), len);
312  array_store_.remove_prefix(s);
313  return true;
314  }
315 
316  if (block_buffer_.size() >= RecordHeader::kSingleSmallSize) {
317  StringPiece cur_rec;
318  unsigned type = ReadPhysicalRecord(&cur_rec);
319  switch (type) {
320  case kFullType:
321  if (in_fragmented_record) {
322  wrapper_->ReportCorruption(scratch->size(), "partial record without end(1)");
323  } else {
324  *record = cur_rec;
325  return true;
326  }
327  /* code */
328  break;
329 
330  default:
331  break;
332  }
333  }
334  }
335  return true;
336 }

338 unsigned ReaderImpl::ReadPhysicalRecord(StringPiece* dest) {
339  RecordHeader rh;
340  size_t parsed = rh.Parse(block_buffer_.data());
341  if (parsed + rh.size > block_buffer_.size()) {
342  wrapper_->ReportCorruption(block_buffer_.size(), "Bad record size");
343  block_buffer_.clear();
344  return kBadRecord;
345  }
346 
347  block_buffer_.advance(parsed);
348  DCHECK_EQ(0, rh.flags & kCompressedFlag);
349  *dest = StringPiece(strings::charptr(block_buffer_.data()), rh.size);
350  block_buffer_.remove_prefix(rh.size);
351 
352  return rh.flags & kTypeMask;
353 }
