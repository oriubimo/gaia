36 void FiberQueueThreadPool::Shutdown() {
37  if (!workers_)
38  return;
39 
40  is_closed_.store(true, std::memory_order_seq_cst);
41  for (size_t i = 0; i < worker_size_; ++i) {
42  workers_[i].pull_ec.notifyAll();
43  }
44  Func f;
45 
46  for (size_t i = 0; i < worker_size_; ++i) {
47  auto& w = workers_[i];
48  pthread_join(w.tid, nullptr);
49  CHECK(!w.q->try_dequeue(f));
50  }

54 }
