26 /*
27  Single producer high performant Task Pool, designed to route work to worker threads.
28  Unlike usualy thread-pool this one has non-trivial life-cycle, allowing setting up local,
29  per thread data for each worker thread and SharedData, accessible by all threads.
30  All this is configured by a single class that encapsulates user logic and life-cycle of the
31  data for that Task.
32 
33  Example:
34  struct MyTask {
35  struct SharedData {
36  mutex* m;
37  Output* o;
38  };
39 
40  void operator(const string& str, SharedData& shared) {
41  // Do something with str.
42  ++i;
43 
44  std::lock_guard<mutex> lock(*shared.m);
45  o->Write(...);
46  }
47 
48  //
49  int i = 0; // Local thread-data, accessible only to this task.
50  };
51 
52  // This TaskPool receives string stream as its input load. every string item is rerouted to
53  // available worker in the pool. The worker can process the string and update shared data.
54  // Private variables of the task are thread-local, can not be shared between tasks.
55  using TaskPool = util::SingleProducerTaskPool<MyTask, string>;
56 
57  // Declare pool with number of workers=O(#cpus), for each worker SPSC queue of size 10.
58  TaskPool pool("pool", 10);
59 
60  pool.SetSharedData(&mutex, &output); // Assign shared data.
61  pool.Launch();
62  ...
63  pool.RunTask(some_string1);
64  pool.RunTask(some_string2);
65  ....
66  pool.WaitForTasksToComplete();
67 
68  // Optionally call Finalize() for tasks that have this function.
69  pool.Finalize()
70 
71 */

246  // Nonblocking routine. Returns true in case it succeeds to add the task to the pool.
247  template <typename... Args>
248  bool TryRunTask(Args&&... args) {
249  unsigned index = FindMostFreeThread();
250 
251  QueueTaskImpl* t = static_cast<QueueTaskImpl*>(thread_interfaces_[index].get());
252 
253  if (t->queue_.write(std::forward<Args>(args)...)) {
254  auto& ti = thread_info_[index];
255  ti.Wake();
256 
257  return true;
258  }
259 
260  return false;
261  }

281  template <typename... Args>
282  void Launch(Args&&... args) {
283  if (calling_thread_task_)
284  return;
285  calling_thread_task_.reset(new Task(std::forward<Args>(args)...));
286  InitShared(*calling_thread_task_, shared_data_);
287 
288  thread_interfaces_.resize(thread_count());
289  for (auto& ti : thread_interfaces_) {
290  ti.reset(new QueueTaskImpl(per_thread_capacity_, shared_data_, std::forward<Args>(args)...));
291  }
292  LaunchThreads();
293  }
294 
295  // Runs sequentially, thread-safe.
296  void Finalize() {
297  if (!calling_thread_task_)
298  return;
299  calling_thread_task_->Finalize();
300  for (auto& ti : thread_interfaces_) {
301  static_cast<QueueTaskImpl*>(ti.get())->Finalize();
302  }
303  }

307  SharedTuple shared_data_;
