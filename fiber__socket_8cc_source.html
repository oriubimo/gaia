48 void FiberSocketImpl::Shutdown(error_code& ec) {
49  if (!is_open_)
50  return;
51  auto handle = sock_.native_handle();
52  auto cb = [&] {
53  if (!is_open_) {
54  DVLOG(1) << "Already closed " << handle;
55  return;
56  }
57 
58  is_open_ = false;
59  sock_.cancel(ec);
60  sock_.shutdown(socket_t::shutdown_both, ec);
61  VLOG(1) << "Sock Shutdown " << handle;
62  if (clientsock_data_) {
63  DVLOG(1) << "Sock Closed";
64  clientsock_data_->worker_cv.notify_one();
65  if (clientsock_data_->worker.joinable())
66  clientsock_data_->worker.join();
67  DVLOG(1) << "Worker Joined";
68  }
69  };
70 
71  if (clientsock_data_) {
72  VLOG(1) << "AwaitShutdown " << handle;
73  clientsock_data_->io_cntx->AwaitSafe(cb);
74  } else {
75  cb();
76  }
77 }

103 void FiberSocketImpl::Worker(const std::string& hname, const std::string& service) {
104  while (is_open_) {
105  if (status_) {
106  error_code ec = Reconnect(hname, service);
107  VLOG(1) << "After Reconnect: " << ec << "/" << ec.message() << " is_open: " << is_open_;
108  if (ec && is_open_) { // Only sleep for open socket for the next reconnect.
109  this_fiber::sleep_for(10ms);
110  }
111  continue;
112  }
113  DCHECK(sock_.non_blocking());
114 
115  error_code ec;
116  VLOG(2) << "BeforeAsyncWait";
117  sock_.async_wait(socket_t::wait_read, fibers_ext::yield[ec]);
118  if (ec) {
119  LOG_IF(ERROR, is_open_) << "AsyncWait: " << ec.message();
120  continue;
121  }
122 
123  size_t read_capacity = rbuf_size_ - rslice_.size();
124  if (read_state_ == READ_IDLE && read_capacity) {
125  uint8_t* next = static_cast<uint8_t*>(rslice_.data()) + rslice_.size();
126  // Direct but non-blocking call since we know we should be able to receive.
127  // Since it's direct - we do not context-switch.
128  size_t read_cnt = sock_.receive(asio::mutable_buffer(next, read_capacity), 0, status_);
129  if (!status_) {
130  rslice_ = asio::mutable_buffer(rslice_.data(), rslice_.size() + read_cnt);
131  } else if (status_ == system::errc::resource_unavailable_try_again) {
132  status_.clear();
133  } else {
134  VLOG(1) << "SockReceive: " << status_.message();
135  }
136  continue;
137  }
138  VLOG(2) << "BeforeCvReadWait";
139  auto should_iterate = [this] {
140  return !is_open() || (read_state_ == READ_IDLE && rslice_.size() != rbuf_size_);
141  };
142 
143  fibers::mutex mu;
144  std::unique_lock<fibers::mutex> lock(mu);
145  clientsock_data_->worker_cv.wait(lock, should_iterate);
146 
147  VLOG(2) << "WorkerIteration: ";
148  }
149  VLOG(1) << "FiberSocketReadExit";
150 }

152 system::error_code FiberSocketImpl::Reconnect(const std::string& hname,
153  const std::string& service) {
154  DCHECK(clientsock_data_);
155  using namespace asio::ip;
156 
157  auto& asio_io_cntx = clientsock_data_->io_cntx->raw_context();
158  // asio::steady_timer timer(asio_io_cntx, clientsock_data_->connect_duration);
159 
160  tcp::resolver resolver(asio_io_cntx);
161 
162  /*timer.async_wait([&](const system::error_code& ec) {
163  if (!ec) { // Successfully expired.
164  VLOG(1) << "Cancelling resolver";
165  resolver.cancel();
166  }
167  });*/
168 
169  system::error_code ec;
170  VLOG(1) << "Before AsyncResolve";
171 
172  // It seems that resolver waits for 10s and ignores cancel command.
173  auto results = resolver.async_resolve(tcp::v4(), hname, service, fibers_ext::yield[ec]);
174  if (ec) {
175  return ec;
176  }
177  DVLOG(1) << "After AsyncResolve";
178 
179  asio::steady_timer timer(asio_io_cntx, clientsock_data_->connect_duration);
180  timer.async_wait([&](const system::error_code& ec) {
181  if (!ec) { // Successfully expired.
182  VLOG(1) << "Cancelling sock_";
183  sock_.cancel();
184  }
185  });
186 
187  asio::async_connect(sock_, results, fibers_ext::yield[ec]);
188  DVLOG(1) << "After async_connect " << ec;
189 
190  if (!ec) {
191  sock_.non_blocking(true); // For some reason async_connect clears this option.
192 
193  // Use mutex to so that WaitToConnect would be thread-safe.
194  std::lock_guard<fibers::mutex> lock(clientsock_data_->connect_mu);
195  status_ = ec;
196  // notify_one awakes only those threads that already suspend on cnd.wait(). Therefore
197  // we must change status_ under mutex.
198  clientsock_data_->cv_st.notify_one();
199  }
200  status_ = ec;
201 
202  return status_;
203 }
