145 auto GCS::SeqReadFile::Drain(SslStream* stream, ::boost::beast::flat_buffer* tmp_buf)
146  -> error_code {
147  uint8_t buf[512];
148  auto& body = parser.get().body();
149  error_code ec;
150 
151  // Drain pending response completely to allow reusing the current connection.
152  while (!parser.is_done()) {
153  body.data = buf;
154  body.size = sizeof(buf);
155  h2::read(*stream, *tmp_buf, parser, ec);
156  if (ec && ec != h2::error::need_buffer) {
157  return ec;
158  }
159  }
160  return ec;
161 }

205 auto GCS::ListBuckets() -> ListBucketResult {
206  RETURN_IF_ERROR(PrepareConnection());
207 
208  string url = absl::StrCat("/storage/v1/b?project=", gce_.project_id());
209  absl::StrAppend(&url, "&fields=items,nextPageToken");

211  auto http_req = PrepareRequest(h2::verb::get, url, access_token_header_);
212 
213  // TODO: to have a handler extracting what we need.
214  rj::Document doc;
215  vector<string> results;
216 
217  while (true) {
218  h2::response<h2::dynamic_body> resp_msg;
219 
220  RETURN_IF_ERROR(HttpMessage(&http_req, &resp_msg));
221  if (resp_msg.result() != h2::status::ok) {
222  return Status(StatusCode::IO_ERROR, absl::StrCat("Http error: ", resp_msg.result_int(), " ",
223  absl_sv(resp_msg.reason())));

225  http::RjBufSequenceStream is(resp_msg.body().data());
226 
227  doc.ParseStream<rj::kParseDefaultFlags>(is);
228  if (doc.HasParseError()) {
229  LOG(ERROR) << rj::GetParseError_En(doc.GetParseError()) << resp_msg;
230  return Status(StatusCode::PARSE_ERROR, "Could not parse json response");
231  }
232 
233  auto it = doc.FindMember("items");
234  CHECK(it != doc.MemberEnd()) << resp_msg;
235  const auto& val = it->value;
236  CHECK(val.IsArray());
237  auto array = val.GetArray();
238 
239  for (size_t i = 0; i < array.Size(); ++i) {
240  const auto& item = array[i];
241  auto it = item.FindMember("name");
242  if (it != item.MemberEnd()) {
243  results.emplace_back(it->value.GetString(), it->value.GetStringLength());
244  }
245  }
246  it = doc.FindMember("nextPageToken");
247  if (it == doc.MemberEnd()) {
248  break;
249  }
250  absl::string_view page_token{it->value.GetString(), it->value.GetStringLength()};
251  http_req.target(absl::StrCat(url, "&pageToken=", page_token));
252  }
253  return results;

334 auto GCS::OpenSequential(absl::string_view bucket, absl::string_view obj_path) -> OpenSeqResult {
335  string read_obj_url = BuildGetObjUrl(bucket, obj_path);
336  auto req = PrepareRequest(h2::verb::get, read_obj_url, access_token_header_);
337  error_code ec;
338  unique_ptr<SeqReadFile> tmp_file;
339  size_t file_size = 0;
340  size_t sleep_sec = 1;
341  unsigned unautharized_iters = 0;
342  unsigned throttle_iters = 0;
343  unsigned error_iters = 0;
344  while (true) {
345  RETURN_IF_ERROR(PrepareConnection());
346 
347  VLOG(1) << "Req: " << req;
348  h2::write(*client_, req, ec);
349  RETURN_EC_STATUS(ec);
350 
351  tmp_file.reset(new SeqReadFile);
352 
353  h2::read_header(*client_, tmp_buffer_, tmp_file->parser, ec);
354  const auto& msg = tmp_file->parser.get();
355  if (ec) {
356  LOG(ERROR) << "Socket iter/error " << error_iters << "/" << ec << " " << ec.message();
357  VLOG(1) << "FiberSocket status: " << client_->next_layer().status();
358 
359  Status st = InitSslClient();
360  if (!st.ok()) {
361  LOG(ERROR) << "Could not reconnect " << st;
362  return st;
363  }
364  ++error_iters;
365  continue;
366  }
367 
368  VLOG(1) << "HeaderResp: " << msg;
369  VLOG(1) << "Msg Accessors: KA" << tmp_file->parser.keep_alive() << "/"
370  << tmp_file->parser.need_eof();
371  if (!tmp_file->parser.keep_alive()) {
372  VLOG(1) << "No keep-alive, requested reconnect";
373  reconnect_needed_ = true;
374  }
375  if (ShouldRetry(msg.result())) {
376  ++throttle_iters;
377  RETURN_EC_STATUS(tmp_file->Drain(client_.get(), &tmp_buffer_));
378  LOG_IF(WARNING, throttle_iters > 2) << "Retrying iteration " << throttle_iters;
379  VLOG(1) << "Retrying iteration " << throttle_iters;
380  this_fiber::sleep_for(chrono::seconds(sleep_sec));
381  sleep_sec = sleep_sec >= 16 ? 16 : sleep_sec * 2;
382  continue;
383  }
384 
385  if (msg.result() == h2::status::ok) {
386  auto content_len_it = msg.find(h2::field::content_length);
387  if (content_len_it != msg.end()) {
388  absl::SimpleAtoi(absl_sv(content_len_it->value()), &file_size);
389  }
390  break;
391  }
392 
393  if (IsUnauthorized(msg)) {
394  if (++unautharized_iters > 1) {
395  return Status(StatusCode::INTERNAL_ERROR, "Access denied");
396  }
397 
398  RETURN_EC_STATUS(tmp_file->Drain(client_.get(), &tmp_buffer_));
399  RETURN_IF_ERROR(RefreshToken(&req));
400  continue;
401  }
402  LOG(ERROR) << "Unsupported error in response " << msg << "/" << msg.reason();
403  return Status(StatusCode::INTERNAL_ERROR, "HTTP ERROR");
404  }
405  seq_file_ = std::move(tmp_file);
406 
407  return file_size;
408 }

411  CHECK(seq_file_ && client_);
412 
413  if (seq_file_->parser.is_done()) {
414  return 0;
415  }
416

496 template <typename RespBody> Status GCS::HttpMessage(Request* req, Response<RespBody>* resp) {
497  for (unsigned i = 0; i < 2; ++i) {
498  VLOG(1) << "Req: " << *req;
499 
500  error_code ec = WriteAndRead(req, resp);
501  if (ec) {
502  return ToStatus(ec);
503  }
504  VLOG(1) << "Resp: " << *resp;
505 
506  if (!IsUnauthorized(*resp)) {
507  break;
508  }
509  RETURN_IF_ERROR(RefreshToken(req));
510  *resp = Response<RespBody>{};
511  }
512  return Status::OK;
