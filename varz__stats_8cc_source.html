75 auto VarzMapCount::ReadLockAndFindOrInsert(StringPiece key) -> Map::iterator {
76  rw_spinlock_.lock_shared();
77  auto it = map_counts_.find(key);
78  if (it != map_counts_.end())
79  return it;
80 
81  rw_spinlock_.unlock_shared();
82 
83  rw_spinlock_.lock();
84  auto res = map_counts_.emplace(key, base::atomic_wrapper<long>(0));
85  rw_spinlock_.unlock_and_lock_shared();
86  return res.first;
87 }

115 VarzValue VarzMapCount::GetData() const {
116  AnyValue::Map result;
