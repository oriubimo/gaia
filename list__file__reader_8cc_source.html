60 bool Lst1Impl::ReadHeader(std::map<std::string, std::string>* dest) {
61  list_file::HeaderParser parser;
62  Status status = parser.Parse(wrapper_->file, dest);
63 
64  if (!status.ok()) {
65  wrapper_->BadHeader(status);
66  return false;
67  }
68 
69  file_offset_ = wrapper_->read_header_bytes = parser.offset();
70  wrapper_->block_size = parser.block_multiplier() * list_file::kBlockSizeFactor;
71 
72  CHECK_GT(wrapper_->block_size, 0);
73  backing_store_.reset(new uint8[wrapper_->block_size]);
74  uncompress_buf_.reset(new uint8[wrapper_->block_size]);
75  if (file_offset_ >= wrapper_->file->Size()) {
76  wrapper_->eof = true;
77  }
78  return true;

81 bool Lst1Impl::ReadRecord(StringPiece* record, std::string* scratch) {
82  scratch->clear();
83  *record = StringPiece();
84  bool in_fragmented_record = false;
85  StringPiece fragment;
86  using namespace list_file;
87 
88  while (true) {
89  if (array_records_ > 0) {
90  uint32 item_size = 0;
91  const uint8* aend = reinterpret_cast<const uint8*>(array_store_.end());
92  const uint8* item_ptr = Varint::Parse32WithLimit(u8ptr(array_store_), aend, &item_size);
93  DVLOG(2) << "Array record with size: " << item_size;
94 
95  const uint8* next_rec_ptr = item_ptr + item_size;
96  if (item_ptr == nullptr || next_rec_ptr > aend) {
97  wrapper_->ReportCorruption(array_store_.size(), "invalid array record");
98  array_records_ = 0;
99  } else {
100  wrapper_->read_header_bytes += item_ptr - u8ptr(array_store_);
101  array_store_.remove_prefix(next_rec_ptr - u8ptr(array_store_));
102  *record = StringPiece(strings::charptr(item_ptr), item_size);
103  wrapper_->read_data_bytes += item_size;
104  --array_records_;
105  return true;
106  }
107  }
108  const unsigned int record_type = ReadPhysicalRecord(&fragment);
109  switch (record_type) {
110  case kFullType:
111  if (in_fragmented_record) {
112  wrapper_->ReportCorruption(scratch->size(), "partial record without end(1)");
113  } else {
114  scratch->clear();
115  *record = fragment;
116 
117  return true;
118  }
119  ABSL_FALLTHROUGH_INTENDED;
120  case kFirstType:
121  if (in_fragmented_record) {
122  wrapper_->ReportCorruption(scratch->size(), "partial record without end(2)");
123  }
124  *scratch = AsString(fragment);
125  in_fragmented_record = true;
126 
127  break;
128 
129  case kMiddleType:
130  if (!in_fragmented_record) {
131  wrapper_->ReportCorruption(fragment.size(), "missing start of fragmented record(1)");
132  } else {
133  scratch->append(fragment.data(), fragment.size());
134  }
135  break;
136 
137  case kLastType:
138  if (!in_fragmented_record) {
139  wrapper_->ReportCorruption(fragment.size(), "missing start of fragmented record(2)");
140  } else {
141  scratch->append(fragment.data(), fragment.size());
142  *record = StringPiece(*scratch);
143  wrapper_->read_data_bytes += record->size();
144  // last_record_offset_ = prospective_record_offset;
145  return true;
146  }
147  break;
148  case kArrayType: {
149  if (in_fragmented_record) {
150  wrapper_->ReportCorruption(scratch->size(), "partial record without end(4)");
151  }
152  uint32 array_records = 0;
153  const uint8* array_ptr = Varint::Parse32WithLimit(
154  u8ptr(fragment), u8ptr(fragment) + fragment.size(), &array_records);
155  if (array_ptr == nullptr || array_records == 0) {
156  wrapper_->ReportCorruption(fragment.size(), "invalid array record");
157  } else {
158  wrapper_->read_header_bytes += array_ptr - u8ptr(fragment);
159  array_records_ = array_records;
160  array_store_ = FromBuf(array_ptr, fragment.end() - strings::charptr(array_ptr));
161  VLOG(2) << "Read array with count " << array_records;
162  }
163  } break;
164  case kEof:
165  if (in_fragmented_record) {
166  wrapper_->ReportCorruption(scratch->size(), "partial record without end(3)");
167  scratch->clear();
168  }
169  return false;
170  case kBadRecord:
171  if (in_fragmented_record) {
172  wrapper_->ReportCorruption(scratch->size(), "error in middle of record");
173  in_fragmented_record = false;
174  scratch->clear();
175  }
176  break;
177  default: {
178  char buf[40];
179  snprintf(buf, sizeof(buf), "unknown record type %u", record_type);
180  wrapper_->ReportCorruption((fragment.size() + (in_fragmented_record ? scratch->size() : 0)),
181  buf);
182  in_fragmented_record = false;
183  scratch->clear();
184  }
185  }
186  }
187  return true;
188 }

190 unsigned int Lst1Impl::ReadPhysicalRecord(StringPiece* result) {
191  using list_file::kBlockHeaderSize;
192  while (true) {
193  if (block_buffer_.size() <= kBlockHeaderSize) {
194  if (!wrapper_->eof) {
195  size_t fsize = wrapper_->file->Size();
196  strings::MutableByteRange mbr(backing_store_.get(), wrapper_->block_size);
197  auto res = wrapper_->file->Read(file_offset_, mbr);
198  VLOG(2) << "read_size: " << res.obj << ", status: " << res.status;
199  if (!res.ok()) {
200  wrapper_->ReportDrop(res.obj, res.status);
201  wrapper_->eof = true;
202  return kEof;
203  }
204  block_buffer_.reset(backing_store_.get(), res.obj);
205  file_offset_ += block_buffer_.size();
206  if (file_offset_ >= fsize) {
207  wrapper_->eof = true;
208  }
209  continue;
210  } else if (block_buffer_.empty()) {
211  // End of file
212  return kEof;
213  } else {
214  size_t drop_size = block_buffer_.size();
215  block_buffer_.clear();
216  wrapper_->ReportCorruption(drop_size, "truncated record at end of file");
217  return kEof;
218  }
219  }
220 
221  // Parse the header
222  const uint8* header = block_buffer_.data();
223  const uint8 type = header[8];
224  uint32 length = coding::DecodeFixed32(header + 4);
225  wrapper_->read_header_bytes += kBlockHeaderSize;
226 
227  if (length == 0 && type == list_file::kZeroType) {
228  size_t bs = block_buffer_.size();
229  block_buffer_.clear();
230  // Handle the case of when mistakenly written last kBlockHeaderSize bytes as empty record.
231  if (bs != kBlockHeaderSize) {
232  LOG(ERROR) << "Bug reading list file " << bs;
233  return kBadRecord;
234  }
235  continue;
236  }
237 
238  if (length + kBlockHeaderSize > block_buffer_.size()) {
239  VLOG(1) << "Invalid length " << length << " file offset " << file_offset_
240  << " block size " << block_buffer_.size() << " type " << int(type);
241  size_t drop_size = block_buffer_.size();
242  block_buffer_.clear();
243  wrapper_->ReportCorruption(drop_size, "bad record length or truncated record at eof.");
244  return kBadRecord;
245  }
246 
247  const uint8* data_ptr = header + kBlockHeaderSize;
248  // Check crc
249  if (wrapper_->checksum) {
250  uint32_t expected_crc = crc32c::Unmask(coding::DecodeFixed32(header));
251  // compute crc of the record and the type.
252  uint32_t actual_crc = crc32c::Value(data_ptr - 1, 1 + length);
253  if (actual_crc != expected_crc) {
254  // Drop the rest of the buffer since "length" itself may have
255  // been corrupted and if we trust it, we could find some
256  // fragment of a real log record that just happens to look
257  // like a valid log record.
258  size_t drop_size = block_buffer_.size();
259  block_buffer_.clear();
260  wrapper_->ReportCorruption(drop_size, "checksum mismatch");
261  return kBadRecord;
262  }
263  }
264  uint32 record_size = length + kBlockHeaderSize;
265  block_buffer_.advance(record_size);
266 
267  if (type & list_file::kCompressedMask) {
268  if (!Uncompress(data_ptr, &length)) {
269  wrapper_->ReportCorruption(record_size, "Uncompress failed.");
270  return kBadRecord;
271  }
272  data_ptr = uncompress_buf_.get();
273  }
274 
275  *result = FromBuf(data_ptr, length);
276  return type & 0xF;
277  }

280 bool Lst1Impl::Uncompress(const uint8* data_ptr, uint32* size) {
281  uint8 method = *data_ptr++;
282  VLOG(2) << "Uncompress " << int(method) << " with size " << *size;
283 
284  uint32 inp_sz = *size - 1;
285 
286  UncompressFunction uncompr_func = GetUncompress(list_file::CompressMethod(method));
287 
288  if (!uncompr_func) {
289  LOG(ERROR) << "Could not find uncompress method " << int(method);
290  return false;
291  }
292  size_t uncompress_size = wrapper_->block_size;
293  Status status = uncompr_func(data_ptr, inp_sz, uncompress_buf_.get(), &uncompress_size);
294  if (!status.ok()) {
295  VLOG(1) << "Uncompress error: " << status;
296  return false;
297  }
298 
299  *size = uncompress_size;
300  return true;
301 }

303 const uint8* DecodeString(const uint8* ptr, const uint8* end, string* dest) {
304  if (ptr == nullptr)
305  return nullptr;
306  uint32 string_sz = 0;
307  ptr = Varint::Parse32WithLimit(ptr, end, &string_sz);
308  if (ptr == nullptr || ptr + string_sz > end)
309  return nullptr;
310  const char* str = reinterpret_cast<const char*>(ptr);
311  dest->assign(str, str + string_sz);
312  return ptr + string_sz;
313 }

355 bool ListReader::ReadHeader() {
356  if (impl_)
357  return true;
358  if (wrapper_->eof)
359  return false;
360 
361  static_assert(list_file::kMagicStringSize == lst2::kMagicStringSize, "");
362 
363  uint8 buf[list_file::kMagicStringSize] = {0};
364 
365  auto res = wrapper_->file->Read(0, strings::MutableByteRange(buf, sizeof(buf)));
366  if (!res.ok()) {
367  wrapper_->BadHeader(res.status);
368  return false;
369  }
370 
371  const auto read_buf = strings::FromBuf(buf, sizeof(buf));
372  const StringPiece kLst1Magic(list_file::kMagicString, list_file::kMagicStringSize);
373  if (read_buf == kLst1Magic) {
374  impl_.reset(new Lst1Impl(wrapper_.get()));
375  return impl_->ReadHeader(&meta_);
376  }
377 
378  const StringPiece kLst2Magic(lst2::kMagicString, lst2::kMagicStringSize);
379  if (read_buf == kLst2Magic) {
380  impl_.reset(new lst2::ReaderImpl(wrapper_.get()));
381  return impl_->ReadHeader(&meta_);
382  }

384  wrapper_->BadHeader(Status(StatusCode::PARSE_ERROR, "Invalid header"));
385  return false;

407 Status list_file::HeaderParser::Parse(file::ReadonlyFile* file,
408  std::map<std::string, std::string>* meta) {
409  uint8 buf[2];
410  auto res = file->Read(kMagicStringSize, strings::MutableByteRange(buf, 2));
411  if (!res.ok())
412  return res.status;
413 
414  offset_ = kListFileHeaderSize;
415  if (buf[0] == 0 || buf[1] > 100) {
416  return Status(StatusCode::IO_ERROR, "Invalid header");
417  }
418 
419  unsigned block_factor = buf[0];
420  if (buf[1] == kMetaExtension) {
421  uint8 meta_header[8];
422  auto res = file->Read(offset_, strings::MutableByteRange(meta_header, sizeof meta_header));
423  if (!res.ok())
424  return res.status;
425 
426  offset_ += res.obj;
427 
428  uint32 length = coding::DecodeFixed32(meta_header + 4);
429  uint32 crc = crc32c::Unmask(coding::DecodeFixed32(meta_header));
430 
431  std::unique_ptr<uint8[]> meta_buf(new uint8[length]);
432  res = file->Read(offset_, strings::MutableByteRange(meta_buf.get(), length));
433  if (!res.ok())
434  return res.status;
435  CHECK_EQ(res.obj, length);
436 
437  offset_ += length;
438  uint32 actual_crc = crc32c::Value(meta_buf.get(), length);
439  if (crc != actual_crc) {
440  LOG(ERROR) << "Corrupted meta data " << actual_crc << " vs2 " << crc;
441  return Status("Bad meta crc");
442  }
443 
444  const uint8* end = meta_buf.get() + length;
445  const uint8* ptr = Varint::Parse32WithLimit(meta_buf.get(), end, &length);
446  for (uint32 i = 0; i < length; ++i) {
447  string key, val;
448  ptr = DecodeString(ptr, end, &key);
449  ptr = DecodeString(ptr, end, &val);
450  if (ptr == nullptr) {
451  return Status("Bad meta crc");
452  }
453  meta->emplace(std::move(key), std::move(val));
454  }
455  }
456  block_multiplier_ = block_factor;
457 
458  return Status::OK;
459 }
